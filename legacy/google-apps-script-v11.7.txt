function _getActualLastRow(sh, colIndex){
  colIndex = colIndex || 1;
  if(!sh) return 1;
  var vals = sh.getRange(1, colIndex, sh.getMaxRows()).getValues();
  for(var i = vals.length - 1; i >= 0; i--){
    var v = vals[i][0];
    if(v !== "" && v !== null && v !== undefined && String(v).trim() !== "") return i + 1;
  }
  return 1;
}

function _num(v){var n = parseFloat(v); return isNaN(n) ? 0 : n;}
function _int(v){var n = parseInt(v, 10); return isNaN(n) ? 0 : n;}
function _safe(v){return v === null || v === undefined ? "" : v;}
function _str(v){return String(_safe(v));}

function _flat(arr){var r=[];for(var i=0;i<arr.length;i++){if(Array.isArray(arr[i])){for(var j=0;j<arr[i].length;j++){r.push(arr[i][j]);}}else{r.push(arr[i]);}}return r;}

function _safeRowHeights(sh, startRow, count, height){
  if(!sh) return;
  try{
    var maxRows = sh.getMaxRows();
    var needed = startRow + count - 1;
    if(needed > maxRows) sh.insertRowsAfter(maxRows, needed - maxRows);
    var safeCount = Math.max(1, Math.min(count, sh.getMaxRows() - startRow + 1));
    sh.setRowHeights(startRow, safeCount, height);
  }catch(e){}
}

function _ensureRows(sh, needed){
  if(!sh) return;
  try{
    var maxRows = sh.getMaxRows();
    if(needed > maxRows) sh.insertRowsAfter(maxRows, needed - maxRows);
  }catch(e){}
}

function _ensureColumns(sh, needed){
  if(!sh) return;
  try {
    var maxCols = sh.getMaxColumns();
    if (needed > maxCols) sh.insertColumnsAfter(maxCols, needed - maxCols);
  } catch(e){}
}

function _getSettingsMap(){
  var sh = _ss().getSheetByName("SETTINGS");
  if(!sh) return {};
  var lr = _getActualLastRow(sh, 1);
  if(lr < 2) return {};
  var rows = sh.getRange(2, 1, lr - 1, 2).getValues();
  var map = {};
  rows.forEach(function(r){
    var k = String(_safe(r[0])).trim();
    if(k) map[k] = r[1];
  });
  return map;
}

function _getListFromSettings(key, fallback){
  var s = _getSettingsMap();
  var raw = String(_safe(s[key] || "")).trim();
  if(!raw) return fallback.slice();
  var arr = raw.split(",").map(function(x){return x.trim()}).filter(Boolean);
  return arr.length ? arr : fallback.slice();
}

function _getInventoryFormulas(){
  return {
    "O2": '=ARRAYFORMULA(IF(A2:A="","",IFERROR(ROUND((M2:M-N2:N)/M2:M*100,0),0)))',
    "AA2": '=ARRAYFORMULA(IF(A2:A="","",IFERROR(W2:W+X2:X+Y2:Y+Z2:Z+AV2:AV+AW2:AW,0)))',
    // ✅ v15.35 CRITICAL FIX: Returns column is now ALWAYS 0 — admin's
    // dashboard previously went negative because pre-v15.32 Return rows
    // still have qty>0 in TRANSACTIONS!E. Even with new returns logged at
    // qty=0, those legacy rows poison SUMPRODUCT and pull AA-AB negative.
    //
    // Stock & sales accounting is now derived from a SINGLE source of truth:
    // the W..Z + AV..AW (SOLD_*) columns, which we decrement directly when a
    // Return is processed (saveReturnFromForm → soldCol Math.max(0, cS-qty)).
    //
    // What this fixes:
    //   • Total Revenue (AF) = N × AA — never negative
    //   • Net Profit  (AJ) = (N-L) × AA - AdSpend — only negative if ad
    //     spend > 0 with no sales (real loss, not a math glitch)
    //   • Dashboard's totalRev / totalNet stay sane after returns
    //
    // TRANSACTIONS still keeps Return rows for audit trail / Telegram
    // notifications — the marker qty=0 from v15.32 is unchanged.
    "AB2": '=ARRAYFORMULA(IF(A2:A="","",0))',
    "AC2": '=ARRAYFORMULA(IF(A2:A="","",IFERROR((S2:S+T2:T+U2:U+V2:V+AT2:AT+AU2:AU)-(AA2:AA-AB2:AB),0)))',
    "AD2": '=ARRAYFORMULA(IF(A2:A="","",IFERROR(S2:S+T2:T+U2:U+V2:V+AT2:AT+AU2:AU,0)))',
    "AE2": '=ARRAYFORMULA(IF(A2:A="","",IFERROR(L2:L*(S2:S+T2:T+U2:U+V2:V+AT2:AT+AU2:AU),0)))',
    "AF2": '=ARRAYFORMULA(IF(A2:A="","",IFERROR(N2:N*(AA2:AA-AB2:AB),0)))',
    "AG2": '=ARRAYFORMULA(IF(A2:A="","",IFERROR(IF(AE2:AE-AF2:AF>0,AE2:AE-AF2:AF,0),0)))',
    "AH2": '=ARRAYFORMULA(IF(A2:A="","",IFERROR((N2:N-L2:L)*(AA2:AA-AB2:AB),0)))',
    "AI2": '=MAP(A2:A,LAMBDA(n,IF(n="","",IFERROR(SUMPRODUCT((AD_TRACKER!B$2:B$2000=n)*AD_TRACKER!C$2:C$2000),0))))',
    "AJ2": '=ARRAYFORMULA(IF(A2:A="","",IFERROR(AH2:AH-AI2:AI,0)))',
    "AK2": '=ARRAYFORMULA(IF(A2:A="","",IF(P2:P<>"Normal",IFERROR((M2:M-N2:N)*(AA2:AA-AB2:AB),0),0)))'
  };
}

function _restoreInventoryFormulas(inv){
  var formulas = _getInventoryFormulas();
  Object.keys(formulas).forEach(function(cell){
    var r = inv.getRange(cell);
    // ✅ v15.35: AB2 (Returns) was previously a SUMPRODUCT counting old
    // Return rows with qty>0 — that's why dashboard went negative even
    // after our v15.32 fix. We MUST force-replace AB2 every restore so
    // the new "always 0" formula takes effect on existing sheets too.
    // For all other cells, only fill in if blank (legacy behavior).
    if (cell === 'AB2') {
      r.setFormula(formulas[cell]);
    } else if (!r.getFormula()) {
      r.setFormula(formulas[cell]);
    }
  });
}

function _hdr(sheet, headers, bg){
  _ensureColumns(sheet, headers.length);
  sheet.getRange(1, 1, 1, headers.length).setValues([headers]);
  sheet.getRange(1, 1, 1, headers.length)
    .setBackground(bg || C.DARK)
    .setFontColor(C.WHITE)
    .setFontWeight("bold")
    .setFontFamily(C.FONT)
    .setFontSize(10)
    .setHorizontalAlignment("center")
    .setVerticalAlignment("middle");
  sheet.setRowHeight(1, 40);
  sheet.setFrozenRows(1);
}

function _logActivity(product, oldSt, newSt){

  sh.getRange(next, 1, 1, 4).setValues([[new Date(), product, oldSt, newSt]]);
}

function _buildOptions(arr, selected){
  return arr.map(function(x){
    var v = String(_safe(x));
    var sel = (selected !== undefined && v === selected) ? " selected" : "";
    return '<option value="' + v.replace(/"/g, '&quot;') + '"' + sel + '>' + (v || "—") + '</option>';
  }).join("");
}

function _getDeliveryCharges(){
  var ss=_ss();
  var sh=ss.getSheetByName("DELIVERY_CHARGES");
  if(!sh){
    _setupDeliveryCharges();
    sh=ss.getSheetByName("DELIVERY_CHARGES");
  }
  var lr=_getActualLastRow(sh,1);
  if(lr<2) return [];
  var rows=sh.getRange(2,1,lr-1,4).getValues();
  return rows.filter(function(r){return r[0] || r[1];}).map(function(r,idx){
    return {
      id:_str(r[0] || ("zone_"+(idx+1))),
      name:_str(r[1]),
      charge:_num(r[2]),
      active:String(r[3]).toLowerCase() !== "false"
    };
  }).filter(function(x){return x.name && x.active;});
}

function _webUpdateDeliveryCharges(body){
  try{
    var sh=_ss().getSheetByName("DELIVERY_CHARGES");
    if(!sh){ _setupDeliveryCharges(); sh=_ss().getSheetByName("DELIVERY_CHARGES"); }
    var locations = body.locations || [];
    if(typeof locations === "string"){try{locations=JSON.parse(locations||"[]");}catch(e){locations=[];}}
    if(!Array.isArray(locations)) locations = [];
    var clean = locations.map(function(loc,idx){
      return [
        _str(loc.id || ("zone_"+(idx+1))).replace(/\s+/g,"_"),
        _str(loc.name || loc.location || ""),
        _num(loc.charge || loc.fee || 0),
        loc.active === undefined ? true : !(String(loc.active).toLowerCase()==="false" || String(loc.active)==="0")
      ];
    }).filter(function(r){return r[1];});
    // ✅ v9.3: Narayanganj defaults if admin sends empty list
    if(!clean.length) clean = [["inside_narayanganj","Inside Narayanganj",70,true],["outside_narayanganj","Outside Narayanganj",140,true]];
    var lr=_getActualLastRow(sh,1);
    if(lr>=2) sh.getRange(2,1,lr-1,4).clearContent();
    _ensureRows(sh, clean.length+1);
    sh.getRange(2,1,clean.length,4).setValues(clean);
    notifyCloudflare(['delivery_charges', 'store_info']);
    return {ok:true,success:true,locations:clean.length};
  }catch(e){return {ok:false,success:false,msg:e.message};}
}

// ✅ v11.2: SELF-TEST for S/3XL — call action="diagnoseS3XL"
// Writes a known test pattern to row 2's S/3XL stock cells, reads back,
// and reports exactly what the deployed code can/cannot do.
// ════════════════════════════════════════════════════════
function diagnoseS3XL(){
  try{
    var report = {
      ok: true,
      version: "v11.2",
      checks: {},
      verdict: ""
    };
    var inv = _ss().getSheetByName("INVENTORY");
    if(!inv){ report.ok=false; report.verdict="INVENTORY sheet missing"; return report; }

    // 1. COL constants — confirm S=46, 3XL=47, SOLD_S=48, SOLD_3XL=49
    report.checks.COL_STK_S = COL.STK_S;
    report.checks.COL_STK_3XL = COL.STK_3XL;
    report.checks.COL_SOLD_S = COL.SOLD_S;
    report.checks.COL_SOLD_3XL = COL.SOLD_3XL;
    var colsOk = (COL.STK_S===46 && COL.STK_3XL===47 && COL.SOLD_S===48 && COL.SOLD_3XL===49);
    report.checks.colsOk = colsOk;

    // 2. Sheet has at least 49 columns
    var maxCols = inv.getMaxColumns();
    report.checks.maxColumns = maxCols;
    if(maxCols < 49){
      _ensureColumns(inv, 49);
      report.checks.afterEnsureCols = inv.getMaxColumns();
    }

    // 3. Headers — what's actually in row 1, cols 46-49
    var hdrRow = inv.getRange(1, 1, 1, 49).getValues()[0];
    report.checks.headers_46_to_49 = [hdrRow[45], hdrRow[46], hdrRow[47], hdrRow[48]];

    // 4. Try writing a sentinel value to row 2 (or first product row) at cols AT/AU
    var lr = _getActualLastRow(inv, 1);
    if(lr < 2){
      report.ok=false; report.verdict="No products in INVENTORY — add one first to test";
      return report;
    }
    var testRow = 2;
    var prodName = inv.getRange(testRow, COL.NAME).getValue();
    report.checks.testRow = testRow;
    report.checks.testProduct = prodName;

    // Read existing values
    var origS  = inv.getRange(testRow, COL.STK_S).getValue();
    var orig3X = inv.getRange(testRow, COL.STK_3XL).getValue();
    report.checks.before = { S: origS, "3XL": orig3X };

    // Write sentinel
    var sentinelS = 7777;
    var sentinel3 = 8888;
    inv.getRange(testRow, COL.STK_S).setValue(sentinelS);
    inv.getRange(testRow, COL.STK_3XL).setValue(sentinel3);
    SpreadsheetApp.flush();

    // Read back
    var readS  = inv.getRange(testRow, COL.STK_S).getValue();
    var read3X = inv.getRange(testRow, COL.STK_3XL).getValue();
    report.checks.afterWrite = { S: readS, "3XL": read3X };
    var writeOk = (Number(readS) === sentinelS && Number(read3X) === sentinel3);
    report.checks.writeOk = writeOk;

    // Restore original
    inv.getRange(testRow, COL.STK_S).setValue(origS);
    inv.getRange(testRow, COL.STK_3XL).setValue(orig3X);
    SpreadsheetApp.flush();
    report.checks.restored = {
      S:  inv.getRange(testRow, COL.STK_S).getValue(),
      "3XL": inv.getRange(testRow, COL.STK_3XL).getValue()
    };

    // 5. Verdict
    if(!colsOk){
      report.verdict = "❌ COL constants wrong — code mismatch with sheet layout";
      report.ok = false;
    } else if(!writeOk){
      report.verdict = "❌ Sheet write failed — cells may be protected or formulas blocking AT/AU";
      report.ok = false;
    } else {
      report.verdict = "✅ S/3XL writes are working correctly. If admin save still fails, your deployment URL is pointing at OLD code — redeploy required.";
    }
    return report;
  }catch(e){
    return { ok:false, error: e.message, stack: e.stack || '' };
  }
}

var STEADFAST_BASE_URL = 'https://portal.packzy.com/api/v1';

    if(!sh) return { ok:false, msg:'SETTINGS not found' };

    };

  }catch(e){ return { ok:false, msg:e.message }; }

// Generic Steadfast HTTP request
function _steadfastRequest(path, method, payload){
  var keys = _steadfastKeys();
  if(!keys.apiKey || !keys.secret) {
    return { ok:false, msg:'Steadfast API keys নেই — Settings থেকে যোগ করুন' };
  }
  var url = STEADFAST_BASE_URL + path;
  var options = {
    method: method || 'get',
    contentType: 'application/json',
    headers: {
      'Api-Key': keys.apiKey,
      'Secret-Key': keys.secret,
      'Accept': 'application/json'
    },
    muteHttpExceptions: true
  };
  if(payload) options.payload = JSON.stringify(payload);
  try {
    var res = UrlFetchApp.fetch(url, options);
    var code = res.getResponseCode();
    var txt = res.getContentText();
    var parsed;
    try { parsed = JSON.parse(txt); } catch(e){ parsed = { raw: txt }; }
    return { ok: code >= 200 && code < 300, status: code, data: parsed };
  } catch(e){
    return { ok:false, msg:e.message };
  }
}

// Create a single Steadfast consignment from one website order
function steadfastCreateOrder(body){
  try{
    var orderId = String(body.orderId || '').trim();
    if(!orderId) return { ok:false, msg:'orderId required' };

    var sh = _ss().getSheetByName('Website_Orders');
    if(!sh) return { ok:false, msg:'Website_Orders not found' };
    var lr = _getActualLastRow(sh, 1);
    if(lr < 2) return { ok:false, msg:'no orders' };
    var ids = _flat(sh.getRange(2,1,lr-1,1).getValues());
    var hits = [];
    ids.forEach(function(id, i){ if(String(id) === orderId) hits.push(i + 2); });
    if(!hits.length) return { ok:false, msg:'order not found' };

    // Aggregate multi-row cart into one consignment
    var rows = hits.map(function(r){ return sh.getRange(r,1,1,19).getValues()[0]; });
    var first = rows[0];
    var customer = String(first[2] || '').substring(0, 100); // Steadfast: max 100 chars
    var phone    = _normalizeBdPhoneServer(String(first[3] || ''));
    var address  = String(first[4] || '');
    var location = String(first[5] || '');
    var customerNotes = String(first[13] || ''); // Column N: customer's actual notes
    var paymentMethod = String(first[12] || 'COD').trim();

    // ✅ v11.6: Sanitize address — strip noisy `| City: ...` `[TrxID: ...]` `[Note: ...]` `[Custom: ...]` tokens
    address = _sanitizeAddressForCourier(address);

    // ✅ v11.6: Sanitize note — strip internal markers (Email/Coupon/TrxID) so courier sees only delivery instructions
    var emailMatch = customerNotes.match(/Email:\s*([^\s|]+)/i);
    var customerEmail = emailMatch ? emailMatch[1] : '';
    var cleanNote = _sanitizeNoteForCourier(customerNotes);

    // ✅ v11.6: Pull bracketed `[Note: ...]` from address into cleanNote if missing
    var addrNoteMatch = String(first[4] || '').match(/\[Note:\s*([^\]]+)\]/i);
    if(addrNoteMatch && !cleanNote) cleanNote = addrNoteMatch[1].trim();

    // Customer name fallback (avoid empty name → Steadfast 422)
    if(!customer) customer = phone || orderId;

    var totalAmount = 0;
    var totalLot = 0;
    var itemDescription = rows.map(function(r){
      var prod = String(r[6] || '');
      var size = String(r[7] || '');
      if (size.toUpperCase() === 'ONE') size = 'One Size'; // ✅ v16.1 ONE-SIZE: friendly label in notifications
      var qty  = parseInt(r[8]) || 1;
      var price = parseFloat(r[9]) || 0;
      totalLot += qty;
      totalAmount += (qty * price);
      // Add per-row delivery (only first row has it; others = 0)
      totalAmount += parseFloat(r[10]) || 0;
      return prod + (size ? ' (Size: ' + size + ')' : '') + ' x' + qty;
    }).join(', ');

    if(!phone) return { ok:false, msg:'Invalid phone number for order ' + orderId };

    // ✅ v11.5 FIX: COD amount = 0 for prepaid payments (bKash/Nagad/Rocket/Bank/Paid)
    var isPrepaid = /^(bkash|nagad|rocket|bank|paid|prepaid)$/i.test(paymentMethod);
    var codAmount = isPrepaid ? 0 : Math.round(totalAmount);

    var fullAddress = (address + (location ? ', ' + location : '')).substring(0, 250); // Steadfast: max 250 chars

    var payload = {
      invoice: orderId,
      recipient_name: customer,
      recipient_phone: phone,
      recipient_address: fullAddress,
      cod_amount: codAmount,
      // ✅ v11.6 FIX: clean note for courier — only delivery instructions
      note: cleanNote.substring(0, 380),
      item_description: itemDescription.substring(0, 380),
      total_lot: totalLot,
      delivery_type: 0  // 0 = home delivery (default)
    };
    // Add optional email if present and valid
    if(customerEmail && /@/.test(customerEmail)) payload.recipient_email = customerEmail.substring(0, 100);
    // Optional alternative phone from caller body if provided
    if(body.alternativePhone){
      var altPh = _normalizeBdPhoneServer(String(body.alternativePhone));
      if(altPh) payload.alternative_phone = altPh;
    }
    // Allow caller to override delivery_type (e.g. 1 = Point/Hub pickup)
    if(body.deliveryType !== undefined && body.deliveryType !== null && body.deliveryType !== ''){
      payload.delivery_type = parseInt(body.deliveryType, 10) || 0;
    }

    var res = _steadfastRequest('/create_order', 'post', payload);
    if(!res.ok || !res.data) {
      var errMsg = 'Steadfast error: ' + (res.data && (res.data.message || res.data.error) ? (res.data.message || res.data.error) : ('HTTP ' + res.status));
      // Surface validation errors per-field if present
      if(res.data && res.data.errors){
        try { errMsg += ' | ' + JSON.stringify(res.data.errors); } catch(_){}
      }
      return { ok:false, msg: errMsg, status: res.status, data: res.data };
    }
    var consignment = (res.data && res.data.consignment) || res.data;
    var trackingCode = consignment.tracking_code || consignment.trackingCode || '';
    var consignmentId = consignment.consignment_id || '';

    // Update all matching rows with status + courier info
    var bdTime = Utilities.formatDate(new Date(), 'Asia/Dhaka', 'yyyy-MM-dd HH:mm:ss');
    var courierField = 'Steadfast' + (trackingCode ? ' | ' + trackingCode : '');
    hits.forEach(function(row){
      sh.getRange(row, 16).setValue('Picked Up'); // P: Status
      sh.getRange(row, 17).setValue(courierField); // Q: Courier
      sh.getRange(row, 18).setValue(bdTime); // R: Updated
      var prevAct = String(sh.getRange(row, 19).getValue() || '');
      var newAct = 'Steadfast pickup created (' + (trackingCode || consignmentId) + ') @ ' + bdTime;
      sh.getRange(row, 19).setValue(prevAct ? (prevAct + ' | ' + newAct) : newAct);
    });

    notifyCloudflare();
    return {
      ok: true,
      success: true,
      orderId: orderId,
      trackingCode: trackingCode,
      consignmentId: consignmentId,
      codAmount: codAmount,
      isPrepaid: isPrepaid,
      consignment: consignment
    };
  }catch(e){
    return { ok:false, msg:e.message };
  }
}

// Bulk create — uses Steadfast's real bulk endpoint when possible (max ~50/batch).
// Falls back to individual /create_order if bulk endpoint rejects.
function steadfastBulkCreate(body){
  try{
    var orderIds = body.orderIds || [];
    if(!Array.isArray(orderIds) || !orderIds.length) return { ok:false, msg:'orderIds array required' };

    var sh = _ss().getSheetByName('Website_Orders');
    if(!sh) return { ok:false, msg:'Website_Orders not found' };
    var lr = _getActualLastRow(sh, 1);
    if(lr < 2) return { ok:false, msg:'no orders' };
    var ids = _flat(sh.getRange(2,1,lr-1,1).getValues());

    // Build the bulk payload — one entry per orderId (aggregating multi-row carts)
    var dataArr = [];
    var idToHits = {};
    var skippedOrders = []; // ✅ v11.6: capture orders we couldn't include so admin sees the failure
    orderIds.forEach(function(orderId){
      var hits = [];
      ids.forEach(function(id, i){ if(String(id) === String(orderId)) hits.push(i + 2); });
      if(!hits.length){ skippedOrders.push({ orderId: orderId, ok:false, msg:'order not found in sheet' }); return; }
      idToHits[orderId] = hits;
      var rows = hits.map(function(r){ return sh.getRange(r,1,1,19).getValues()[0]; });
      var first = rows[0];
      var phone = _normalizeBdPhoneServer(String(first[3] || ''));
      if(!phone){ skippedOrders.push({ orderId: orderId, ok:false, msg:'invalid phone number' }); return; }
      var customerNotes = String(first[13] || '');
      var paymentMethod = String(first[12] || 'COD').trim();
      var emailMatch = customerNotes.match(/Email:\s*([^\s|]+)/i);
      var customerEmail = emailMatch ? emailMatch[1] : '';
      // ✅ v11.6: sanitize address & note
      var addressClean = _sanitizeAddressForCourier(String(first[4] || ''));
      var cleanNote = _sanitizeNoteForCourier(customerNotes);
      var addrNoteMatch = String(first[4] || '').match(/\[Note:\s*([^\]]+)\]/i);
      if(addrNoteMatch && !cleanNote) cleanNote = addrNoteMatch[1].trim();
      var custName = String(first[2] || '').substring(0, 100);
      if(!custName) custName = phone || String(orderId);
      var totalAmount = 0, totalLot = 0;
      var itemDesc = rows.map(function(r){
        var prod = String(r[6] || '');
        var size = String(r[7] || '');
        if (size.toUpperCase() === 'ONE') size = 'One Size'; // ✅ v16.1 ONE-SIZE: friendly label
        var qty  = parseInt(r[8]) || 1;
        var price = parseFloat(r[9]) || 0;
        totalLot += qty;
        totalAmount += (qty * price) + (parseFloat(r[10]) || 0);
        return prod + (size ? ' (Size: ' + size + ')' : '') + ' x' + qty;
      }).join(', ');
      var isPrepaid = /^(bkash|nagad|rocket|bank|paid|prepaid)$/i.test(paymentMethod);
      var entry = {
        invoice: String(orderId),
        recipient_name: custName,
        recipient_phone: phone,
        recipient_address: (addressClean + (first[5] ? ', ' + first[5] : '')).substring(0, 250),
        cod_amount: isPrepaid ? 0 : Math.round(totalAmount),
        note: cleanNote.substring(0, 380),
        item_description: itemDesc.substring(0, 380),
        total_lot: totalLot,
        delivery_type: 0
      };
      if(customerEmail && /@/.test(customerEmail)) entry.recipient_email = customerEmail.substring(0, 100);
      dataArr.push(entry);
    });

    if(!dataArr.length) return { ok:false, msg:'No valid orders to send (all had invalid phone or were missing)', failures: skippedOrders };

    // Try the real bulk endpoint first — max 50 per call per Steadfast docs
    var results = [];
    var BATCH = 50;
    for(var i=0; i<dataArr.length; i+=BATCH){
      var batch = dataArr.slice(i, i+BATCH);
      var bulkRes = _steadfastRequest('/create_order/bulk-order', 'post', { data: batch });
      if(bulkRes.ok && bulkRes.data){
        // Bulk endpoint returned. Update sheets per response item if available.
        var arr = bulkRes.data.consignments || bulkRes.data.data || (Array.isArray(bulkRes.data) ? bulkRes.data : []);
        if(!arr.length){
          // Endpoint succeeded but returned no per-item data — fall back to per-order so we can update sheet
          for(var j=0; j<batch.length; j++){
            var perRes = steadfastCreateOrder({ orderId: batch[j].invoice });
            results.push(Object.assign({ orderId: batch[j].invoice }, perRes));
            Utilities.sleep(120);
          }
        } else {
          var bdTime = Utilities.formatDate(new Date(), 'Asia/Dhaka', 'yyyy-MM-dd HH:mm:ss');
          arr.forEach(function(c){
            var orderId = String(c.invoice || c.Invoice || '');
            var trackingCode = c.tracking_code || c.trackingCode || '';
            var consignmentId = c.consignment_id || c.consignmentId || '';
            var hits = idToHits[orderId] || [];
            var courierField = 'Steadfast' + (trackingCode ? ' | ' + trackingCode : '');
            hits.forEach(function(row){
              sh.getRange(row, 16).setValue('Picked Up');
              sh.getRange(row, 17).setValue(courierField);
              sh.getRange(row, 18).setValue(bdTime);
              var prevAct = String(sh.getRange(row, 19).getValue() || '');
              var newAct = 'Steadfast pickup created (' + (trackingCode || consignmentId) + ') @ ' + bdTime;
              sh.getRange(row, 19).setValue(prevAct ? (prevAct + ' | ' + newAct) : newAct);
            });
            results.push({ orderId: orderId, ok: true, trackingCode: trackingCode, consignmentId: consignmentId });
          });
        }
      } else {
        // Bulk endpoint failed — fall back to per-order create so we don't lose any
        for(var k=0; k<batch.length; k++){
          var pr = steadfastCreateOrder({ orderId: batch[k].invoice });
          results.push(Object.assign({ orderId: batch[k].invoice }, pr));
          Utilities.sleep(120);
        }
      }
    }

    notifyCloudflare();
    var success = results.filter(function(r){ return r.ok; }).length;
    // ✅ v11.6: include skipped orders so admin sees per-order failure reasons
    var allResults = skippedOrders.concat(results);
    var failures = allResults.filter(function(r){ return !r.ok; });
    return {
      ok:true, success:true,
      total: allResults.length,
      succeeded: success,
      failed: failures.length,
      results: allResults,
      failures: failures
    };
  }catch(e){
    return { ok:false, msg:e.message };
  }
}

// ✅ v11.6: Strip noisy meta tokens from address before sending to courier
// Removes: "| City: ...", "[Note: ...]", "[Custom: ...]", "[TrxID: ...]"
// Also collapses double-spaces and stray separators
function _sanitizeAddressForCourier(raw){
  if(!raw) return '';
  var s = String(raw);
  s = s.replace(/\s*\|\s*City:[^|]*/gi, '');
  s = s.replace(/\s*\[\s*Note:[^\]]*\]/gi, '');
  s = s.replace(/\s*\[\s*Custom[^\]]*\]/gi, '');
  s = s.replace(/\s*\[\s*TrxID:[^\]]*\]/gi, '');
  s = s.replace(/\s*\[\s*Email:[^\]]*\]/gi, '');
  s = s.replace(/\s*\|\s*$|^\s*\|\s*/g, '');
  s = s.replace(/\s{2,}/g, ' ').trim();
  return s;
}

// ✅ v11.6: Strip internal markers from note so courier sees only customer's actual delivery instructions
function _sanitizeNoteForCourier(raw){
  if(!raw) return '';
  var s = String(raw);
  s = s.replace(/Email:\s*[^\s|]+/gi, '');
  s = s.replace(/Applied Coupon:[^|]+/gi, '');
  s = s.replace(/Sender No:[^|]+/gi, '');
  // Strip leading/trailing pipe separators
  s = s.replace(/^[\s|]+|[\s|]+$/g, '');
  s = s.replace(/\s*\|\s*\|\s*/g, ' | ');
  s = s.replace(/\s{2,}/g, ' ').trim();
  return s;
}

// Server-side phone normalizer (mirror of admin _normalizeBdPhone)
function _normalizeBdPhoneServer(raw){
  if(!raw) return '';
  var d = String(raw).replace(/[^0-9]/g, '');
  if(!d) return '';
  if(d.indexOf('880') === 0) d = d.substring(3);
  if(d.charAt(0) === '0') d = d.substring(1);
  if(d.length > 10) d = d.substring(d.length - 10);
  if(d.length !== 10 || d.charAt(0) !== '1') return '';
  return '01' + d.substring(1); // Steadfast accepts 01XXXXXXXXX (no +880)
}

// ===== onEdit Trigger =====
// ════════════════════════════════════════════════════════
function onEdit(e){
  if(!e||!e.range)return;
  var ss=e.source;
  var sh=ss.getActiveSheet();
  var name=sh.getName();
  var row=e.range.getRow();
  var col=e.range.getColumn();
  if(row<2)return;
  if(name==="INVENTORY"){
    if(col===COL.NAME&&e.value&&!e.oldValue){
      if(!sh.getRange(row,COL.STATUS).getValue())sh.getRange(row,COL.STATUS).setValue("Draft");
      return;
    }
    if(SOLD_COLS.indexOf(col)!==-1){
      sh.getRange(row,COL.UPDATED).setValue(new Date());
      var diff=_int(e.value)-_int(e.oldValue);
      if(diff<=0)return;
      var product=sh.getRange(row,COL.NAME).getValue();
      if(!product)return;
      var sale=_num(sh.getRange(row,COL.SALE).getValue());
      var cost=_num(sh.getRange(row,COL.COST).getValue());
      // ✅ v10 FIX: Include S and 3XL in size mapping (was missing)
      var sizeMap={};
      sizeMap[COL.SOLD_S]="S";
      sizeMap[COL.SOLD_M]="M";
      sizeMap[COL.SOLD_L]="L";
      sizeMap[COL.SOLD_XL]="XL";
      sizeMap[COL.SOLD_XXL]="XXL";
      sizeMap[COL.SOLD_3XL]="3XL";
      _logTransaction([new Date(),product,"Sale",sizeMap[col],diff,diff*sale,diff*cost,diff*(sale-cost)]);
      try { notifyCloudflare(); } catch(_){}
      return;
    }
    if(col===COL.STATUS&&e.value&&e.oldValue&&e.oldValue!==e.value){
      var p2=sh.getRange(row,COL.NAME).getValue();
      _logActivity(p2,e.oldValue,e.value);
    }
  }
  if(name==="DRAFT_VIEW"&&col===13&&row>=2){
    var action=e.value;
    var product=sh.getRange(row,2).getValue();
    if(!product||!action)return;
    var inv=ss.getSheetByName("INVENTORY");
    var lr=_getActualLastRow(inv,1);
    var names=_flat(inv.getRange(2,1,lr-1,1).getValues());
    var idx=names.indexOf(product);
    if(idx===-1)return;
    var r=idx+2;
    if(action==="→ Activate"){
      inv.getRange(r,COL.STATUS).setValue("Active");
      inv.getRange(r,COL.UPDATED).setValue(new Date());
      _logActivity(product,"Draft","Active");
    } else if(action==="→ Archive"){
      inv.getRange(r,COL.STATUS).setValue("Archived");
      inv.getRange(r,COL.UPDATED).setValue(new Date());
      _logActivity(product,"Draft","Archived");
    }
    e.range.clearContent();
  }
  if(name==="ARCHIVE_VIEW"&&col===13&&row>=2){
    var action2=e.value;
    var product2=sh.getRange(row,2).getValue();
    if(!product2||!action2)return;
    var inv2=ss.getSheetByName("INVENTORY");
    var lr2=_getActualLastRow(inv2,1);
    var names2=_flat(inv2.getRange(2,1,lr2-1,1).getValues());
    var idx2=names2.indexOf(product2);
    if(idx2===-1)return;
    var r2=idx2+2;
    var newSt=(action2==="→ Activate")?"Active":"Draft";
    inv2.getRange(r2,COL.STATUS).setValue(newSt);
    inv2.getRange(r2,COL.UPDATED).setValue(new Date());
    _logActivity(product2,"Archived",newSt);
    e.range.clearContent();
  }
}

// ✅ saveProductFromForm — writes ALL columns including Img4/5/6 + Coupon
// ════════════════════════════════════════════════════════
function saveProductFromForm(d){
  try{
    var ss=_ss();
    var inv=ss.getSheetByName("INVENTORY");
    if(!inv)return{ok:false,msg:"INVENTORY নেই"};
    if(!d || !d.name)return{ok:false,msg:"নাম দরকার"};
    _ensureColumns(inv, 52);
    var lr=_getActualLastRow(inv,1);
    if(lr>=2){
      var names=_flat(inv.getRange(2,1,lr-1,1).getValues());
      if(names.indexOf(d.name)!==-1)return{ok:false,msg:"এই নামে already আছে"};
    }
    var row=lr+1;
    _ensureRows(inv,row);

    // BUILD row array (1..45) for batch write
    // ✅ v16.1 ONE-SIZE: array grown to 50 so we can persist HIDDEN_SIZES (col
    // 50). For a sizeless product the admin sends d.oneSize=true → we store the
    // single qty in the M slot and write the "__ONESIZE__" sentinel into
    // HIDDEN_SIZES so the storefront knows to skip the size picker.
    // ✅ v16.2: array grown to 51 to also persist SIZE_TYPE (col 51 = AY).
    // ✅ v16.3: array grown to 52 to also persist ACCESSORY (col 52 = AZ).
    var rowVals = new Array(52).fill("");
    rowVals[COL.NAME-1]          = d.name;
    rowVals[COL.IMG-1]           = _str(d.img1);
    rowVals[COL.IMG2-1]          = _str(d.img2);
    rowVals[COL.IMG3-1]          = _str(d.img3);
    rowVals[COL.VIDEO-1]         = _str(d.vid);
    rowVals[COL.DESC-1]          = _str(d.desc);
    rowVals[COL.CATEGORY-1]      = _str(d.cat);
    rowVals[COL.FABRIC-1]        = _str(d.fab);
    rowVals[COL.BADGE-1]         = _str(d.bad);
    rowVals[COL.SIZE_CHART-1]    = _str(d.sc);
    rowVals[COL.DELIVERY_DAYS-1] = _str(d.ddays || "2-3 days");
    rowVals[COL.COST-1]          = _num(d.cost);
    rowVals[COL.REG-1]           = _num(d.reg);
    rowVals[COL.SALE-1]          = _num(d.sale);
    // DISC_PCT (col 15) is formula — skip
    rowVals[COL.DISC_TYPE-1]     = _str(d.dt || "Normal");
    rowVals[COL.DELIVERY_IN-1]   = _num(d.din)||60;
    rowVals[COL.DELIVERY_OUT-1]  = _num(d.dout)||120;
    rowVals[COL.STK_S-1]         = _int(d.sS);
    rowVals[COL.STK_M-1]         = _int(d.sM);
    rowVals[COL.STK_L-1]         = _int(d.sL);
    rowVals[COL.STK_XL-1]        = _int(d.sXL);
    rowVals[COL.STK_XXL-1]       = _int(d.sXXL);
    rowVals[COL.STK_3XL-1]       = _int(d.s3XL);
    rowVals[COL.SOLD_S-1]        = 0;
    rowVals[COL.SOLD_M-1]        = 0;
    rowVals[COL.SOLD_L-1]        = 0;
    rowVals[COL.SOLD_XL-1]       = 0;
    rowVals[COL.SOLD_XXL-1]      = 0;
    rowVals[COL.SOLD_3XL-1]      = 0;
    // 27..37 (TOT_SOLD..DISC_IMPACT) are formulas — skip
    rowVals[COL.UPDATED-1]       = new Date();
    rowVals[COL.STATUS-1]        = _str(d.status || "Draft");

    rowVals[COL.IMG6-1]          = _str(d.img6);
    rowVals[COL.C_ACT-1]         = _str(d.cAct || "No");
    rowVals[COL.C_CODE-1]        = _str(d.cCode);
    rowVals[COL.C_DISC-1]        = _num(d.cDisc);
    // ✅ v16.1 ONE-SIZE: persist the sentinel (or honor an explicit hiddenSizes
    // list if one was sent). Sizeless products get "__ONESIZE__".
    rowVals[COL.HIDDEN_SIZES-1]  = d.oneSize ? "__ONESIZE__" : _str(d.hiddenSizes || "");
    // ✅ v16.2: persist Size Type override (""/auto/shirt/pant).
    rowVals[COL.SIZE_TYPE-1]     = _str(d.sizeType || "");
    // ✅ v16.3: persist Accessory flag ("Yes" / "No").
    rowVals[COL.ACCESSORY-1]     = (d.accessory === true || String(d.accessory).toLowerCase() === "yes") ? "Yes" : "No";

    // Write columns individually (skip formula cols)
    var skip = {15:1, 27:1,28:1,29:1,30:1,31:1,32:1,33:1,34:1,35:1,36:1,37:1};
    for(var c=1; c<=52; c++){
      if(skip[c]) continue;
      inv.getRange(row, c).setValue(rowVals[c-1]);
    }

    _restoreInventoryFormulas(inv);
    _logActivity(d.name, "", _str(d.status || "Draft"));
    notifyCloudflare();
    // ✅ v11.2: Return verified stored S/3XL values so admin can confirm

      S:   _int(inv.getRange(row, COL.STK_S).getValue()),
      M:   _int(inv.getRange(row, COL.STK_M).getValue()),
      L:   _int(inv.getRange(row, COL.STK_L).getValue()),

      "3XL": _int(inv.getRange(row, COL.STK_3XL).getValue())
    };
    return {ok:true, success:true, verify:verify};
  }catch(err){
    return {ok:false, msg:err.message};
  }
}

function applyBulkEdit(d){
  try{
    var inv=_ss().getSheetByName("INVENTORY");
    // ✅ v16.4 BUGFIX: ensure the sheet physically has all 52 columns BEFORE
    // we write HIDDEN_SIZES(50)/SIZE_TYPE(51)/ACCESSORY(52). Every other
    // inventory writer (saveProductFromForm/Edit, _buildPublicData) already
    // does this; applyBulkEdit was the lone exception, so on an older 51-column
    // sheet `getRange(row,52).setValue()` threw "Range outside dimensions" and
    // the ENTIRE bulk edit (status/price/coupon too) silently failed.
    _ensureColumns(inv, 52);
    var lr=_getActualLastRow(inv,1);
    if(lr<2)return{ok:false,msg:"empty"};
    var names=_flat(inv.getRange(2,1,lr-1,1).getValues());
    var targets = d.names || d.products || [];
    var newStatus = d.st || d.status || "";
    var newBadge = (d.bd !== undefined && d.bd !== "") ? d.bd : (d.badge !== undefined ? d.badge : null);
    var newDiscount = parseFloat(d.disc || d.discount || 0);
    var delInside = (d.delInside !== undefined && d.delInside !== null) ? parseFloat(d.delInside) : null;
    var delOutside = (d.delOutside !== undefined && d.delOutside !== null) ? parseFloat(d.delOutside) : null;
    var newCategory = d.category || "";
    // ✅ v11.1: Coupon bulk fields
    var couponActive = (d.couponActive !== undefined && d.couponActive !== "") ? String(d.couponActive) : null;
    var couponCode   = (d.couponCode   !== undefined && d.couponCode   !== "") ? String(d.couponCode)   : null;
    var couponDisc   = (d.couponDisc   !== undefined && d.couponDisc   !== null && d.couponDisc !== "") ? parseFloat(d.couponDisc) : null;
    // ✅ v15.94: Per-product Hidden Sizes (bulk).
    //   d.hiddenSizes can be:
    //     • a string "S,XXL"           → SET (overwrite) the hidden-size list
    //     • "__CLEAR__"                → clear all hidden sizes (show every size)
    //     • undefined / null / ""      → SKIP (leave existing value untouched)
    var hiddenSizesRaw = d.hiddenSizes;
    var hiddenSizesMode = 'skip';
    var hiddenSizesValue = '';
    if (hiddenSizesRaw === '__CLEAR__') {
      hiddenSizesMode = 'clear';
    } else if (typeof hiddenSizesRaw === 'string' && hiddenSizesRaw.trim() !== '') {
      hiddenSizesMode = 'set';
      // Normalize: uppercase, trim, dedupe, comma-join
      var _seen = {};
      hiddenSizesValue = hiddenSizesRaw.split(',')
        .map(function(s){ return String(s).trim().toUpperCase(); })
        .filter(function(s){ if(!s || _seen[s]) return false; _seen[s]=1; return true; })
        .join(',');
    }
    // ✅ v16.2: Per-product Size Type (bulk). The client only includes
    //   d.sizeType in the payload when the owner picked something:
    //     • "shirt" / "pant"  → SET that override
    //     • "" (Auto)         → CLEAR the override (auto = empty/default)
    //   When d.sizeType is undefined (field left on "Skip") → leave untouched.
    var sizeTypeMode = 'skip';            // 'skip' | 'write'
    var sizeTypeValue = '';
    if (d.sizeType !== undefined) {
      sizeTypeMode = 'write';
      var _stRaw = String(d.sizeType).trim().toLowerCase();
      sizeTypeValue = (_stRaw === 'shirt' || _stRaw === 'pant') ? _stRaw : '';
    }
    // ✅ v16.3: Per-product Accessory flag (bulk). The client only includes
    //   d.accessory in the payload when the owner picked something:
    //     • "yes" → flag selected products as accessories
    //     • "no"  → unflag (move back to normal apparel)
    //   When d.accessory is undefined (field left on "Skip") → leave untouched.
    var accessoryMode = 'skip';           // 'skip' | 'write'
    var accessoryValue = 'No';
    if (d.accessory !== undefined) {
      accessoryMode = 'write';
      accessoryValue = (d.accessory === true || String(d.accessory).trim().toLowerCase() === 'yes') ? 'Yes' : 'No';
    }
    targets.forEach(function(name){
      var idx=names.indexOf(name);
      if(idx===-1)return;
      var row=idx+2;
      if(newStatus) inv.getRange(row,COL.STATUS).setValue(newStatus);
      if(newCategory) inv.getRange(row,COL.CATEGORY).setValue(newCategory);
      if(newBadge !== null) inv.getRange(row,COL.BADGE).setValue(newBadge);
      if(delInside !== null) inv.getRange(row,COL.DELIVERY_IN).setValue(delInside);
      if(delOutside !== null) inv.getRange(row,COL.DELIVERY_OUT).setValue(delOutside);
      if(newDiscount > 0){
        var regPrice = parseFloat(inv.getRange(row,COL.REG).getValue()) || 0;
        if(regPrice > 0){
          var newSale = Math.round(regPrice * (1 - newDiscount/100));
          inv.getRange(row,COL.SALE).setValue(newSale);
        }
      }
      // ✅ v11.1: Coupon writes
      if(couponActive !== null) inv.getRange(row,COL.C_ACT).setValue(couponActive);
      if(couponCode   !== null) inv.getRange(row,COL.C_CODE).setValue(couponCode);
      if(couponDisc   !== null) inv.getRange(row,COL.C_DISC).setValue(couponDisc);
      // ✅ v15.94: Hidden Sizes write (set / clear / skip)
      // ✅ v16.1 ONE-SIZE GUARD: never var a Size-Visibility bulk edit overwrite
      // a sizeless product's "__ONESIZE__" sentinel (that would silently turn a
      // cap/watch back into a sized product). Skip the hiddenSizes write for any
      // selected product currently flagged one-size; all its OTHER bulk fields
      // (status/price/coupon/etc.) still apply normally.
      if(hiddenSizesMode === 'set' || hiddenSizesMode === 'clear'){
        var _curHS = String(inv.getRange(row,COL.HIDDEN_SIZES).getValue()||'').trim().toUpperCase();
        if(_curHS !== '__ONESIZE__'){
          inv.getRange(row,COL.HIDDEN_SIZES).setValue(hiddenSizesMode === 'set' ? hiddenSizesValue : '');
        }
      }
      // ✅ v16.2: Size Type write (set 'shirt'/'pant' or clear to '' for Auto).
      if(sizeTypeMode === 'write'){
        inv.getRange(row,COL.SIZE_TYPE).setValue(sizeTypeValue);
      }
      // ✅ v16.3: Accessory flag write (set 'Yes'/'No').
      if(accessoryMode === 'write'){
        inv.getRange(row,COL.ACCESSORY).setValue(accessoryValue);
      }
      inv.getRange(row,COL.UPDATED).setValue(new Date());
    });
    notifyCloudflare();
    return{ok:true,success:true};
  }catch(e){return{ok:false,msg:e.message};}
}

function saveOrderFromForm(d){
  var lock = LockService.getScriptLock();
  try{
    lock.waitLock(30000);
    var ss=_ss();
    var sh=ss.getSheetByName("ORDERS");
    if(!sh)return{ok:false,msg:"ORDERS নেই"};
    var next=_getActualLastRow(sh,1)+1;
    _ensureRows(sh,next);
    var orderId="ORD-"+Date.now();
    var total=(_num(d.price)*_int(d.qty))+_num(d.dlv);
    sh.getRange(next,1,1,16).setValues([[
      new Date(),orderId,d.cust,d.ph,d.addr,d.loc,d.prod,d.sz,d.qty,d.price,d.dlv,total,d.pay,"Pending","",d.notes
    ]]);

    // ✅ v10 FIX: Manual order should also decrement inventory stock.
    // Without this, manual sales never reflect on stock counts and customers
    // can keep ordering products that are physically out of stock.
    try {
      var inv = ss.getSheetByName("INVENTORY");
      if (inv) {
        var lr = _getActualLastRow(inv, 1);
        if (lr >= 2) {
          var names = _flat(inv.getRange(2, 1, lr - 1, 1).getValues());
          var idx = names.indexOf(d.prod);
          if (idx > -1) {
            var soldCol = {S:COL.SOLD_S,M:COL.SOLD_M,L:COL.SOLD_L,XL:COL.SOLD_XL,XXL:COL.SOLD_XXL,"3XL":COL.SOLD_3XL,ONE:COL.SOLD_M}[_str(d.sz).toUpperCase()];
            if (soldCol) {
              var r = idx + 2;
              var cur = _int(inv.getRange(r, soldCol).getValue());
              inv.getRange(r, soldCol).setValue(cur + _int(d.qty));
              inv.getRange(r, COL.UPDATED).setValue(new Date());
              // Log to TRANSACTIONS so financial reports stay accurate
              var sale = _num(inv.getRange(r, COL.SALE).getValue());
              var cost = _num(inv.getRange(r, COL.COST).getValue());
              var qty = _int(d.qty);
              _logTransaction([new Date(), d.prod, "Sale", d.sz, qty, qty*sale, qty*cost, qty*(sale-cost)]);
            }
          }
        }
      }
    } catch(e){ Logger.log('[inventory] decrement error: ' + e.message); }

    try { _updateCustomerLTV(d.ph, d.cust || '', total, new Date(), d.loc || ''); } catch(e){ Logger.log('[LTV] update error: ' + e.message); }

    notifyCloudflare();
    return{ok:true,success:true,orderId:orderId};
  }catch(e){
    return{ok:false,msg: e.message.includes("lock") ? "System very busy. Try again." : e.message};
  }finally{
    if(lock) lock.releaseLock();
  }
}

function saveReturnFromForm(d){
  try{
    var ss=_ss();
    var inv=ss.getSheetByName("INVENTORY");
    var lr=_getActualLastRow(inv,1);
    var sale=0,cost=0;
    var row = -1;
    if(lr>=2){
      var names=_flat(inv.getRange(2,1,lr-1,1).getValues());
      var idx=names.indexOf(d.prod);
      if(idx>-1){
        row=idx+2;
        sale=_num(inv.getRange(row,COL.SALE).getValue());
        cost=_num(inv.getRange(row,COL.COST).getValue());
      }
    }
    var qty=_int(d.qty)||1;

    // ✅ v10 FIX: Cancel matching Sale entries first so today's revenue/profit
    // returns to ZERO when a same-day sale is returned (instead of going negative).
    // We find the most recent un-cancelled Sale entries for this product+size and
    // remove qty units worth, then log the Return as a marker (no negative numbers).
    var tx = ss.getSheetByName("TRANSACTIONS");
    var remaining = qty;
    if (tx) {
      var txLr = _getActualLastRow(tx, 1);
      if (txLr >= 2) {
        var txData = tx.getRange(2, 1, txLr - 1, 8).getValues();
        // Walk most-recent first
        for (var i = txData.length - 1; i >= 0 && remaining > 0; i--) {
          var r = txData[i];
          if (String(r[1]) === String(d.prod) && String(r[2]) === "Sale" &&
              String(r[3]) === String(d.sz) && _int(r[4]) > 0) {
            var rowQty = _int(r[4]);
            var unitRev = _num(r[5]) / rowQty;
            var unitCost = _num(r[6]) / rowQty;
            var take = Math.min(rowQty, remaining);
            var newQty = rowQty - take;
            var rowIdx = i + 2;
            if (newQty <= 0) {
              // Fully cancel this sale entry
              tx.deleteRow(rowIdx);
              // After delete, all subsequent indexes shift up — restart loop
              txData = tx.getRange(2, 1, _getActualLastRow(tx,1) - 1, 8).getValues();
              i = txData.length; // will become length-1 after i--
            } else {
              // Partial cancel — reduce qty and proportional rev/cost/profit
              tx.getRange(rowIdx, 5).setValue(newQty);
              tx.getRange(rowIdx, 6).setValue(unitRev * newQty);
              tx.getRange(rowIdx, 7).setValue(unitCost * newQty);
              tx.getRange(rowIdx, 8).setValue((unitRev - unitCost) * newQty);
            }
            remaining -= take;
          }
        }
      }
    }

    // ✅ v15.32 FIX: Log Return marker with qty=0 in column E.
    // CRITICAL ISSUE before this fix: The INVENTORY sheet's AB2 formula
    //   =SUMPRODUCT((B=prod)*(C="Return")*E)
    // counts the qty column (E) of every Return row. We were logging qty=qty
    // there, then ALSO decrementing soldCol below. Net effect:
    //   AA (sold) - AB (returns) = 0 - qty = -qty
    //   AF (Revenue) = sale × -qty = NEGATIVE
    //   AJ (Net Profit) = (sale-cost) × -qty - ad_spend = NEGATIVE
    // → Dashboard showed Total Revenue ৳-222 and Net Profit ৳-111.
    //
    // Now: qty=0 in column E so AB stays 0 for matched returns. The
    // actual returned units are already accounted for via soldCol
    // decrement below — that single source of truth keeps the math clean.
    //
    // For audit trail readability we keep the size + a descriptive
    // marker in the cost/profit columns (still 0 to not affect totals).
    // Both matched and orphan returns log identically — qty=0 keeps the
    // dashboard clean while the entry still appears in TRANSACTIONS for
    // traceability. Single source of truth for stock movement is the
    // soldCol decrement below.
    _logTransaction([new Date(), d.prod, "Return", d.sz, 0, 0, 0, 0]);

    // Update INVENTORY metrics
    if(row > -1){
      // ✅ v10 FIX: Include S and 3XL in size mapping
      // ✅ v16.1 ONE-SIZE: "ONE" token (sizeless products) → SOLD_M, uppercase
      // lookup so a return for a cap/watch/etc. correctly restores its stock.
      var soldCol={S:COL.SOLD_S,M:COL.SOLD_M,L:COL.SOLD_L,XL:COL.SOLD_XL,XXL:COL.SOLD_XXL,"3XL":COL.SOLD_3XL,ONE:COL.SOLD_M}[_str(d.sz).toUpperCase()];
      if(soldCol){
        var cS = _int(inv.getRange(row, soldCol).getValue());
        inv.getRange(row, soldCol).setValue(Math.max(0, cS - qty));
      }

      // ✅ v10 FIX: Do NOT manually write to AB (Returns) — it's an array formula.
      // The MAP/LAMBDA at AB2 auto-counts Return entries from TRANSACTIONS,
      // and writing here would break the spill formula for ALL products.
    }

    // ✅ v15.35 FIX: Force-restore inventory formulas after every Return so
    // the new AB2='ARRAYFORMULA(...,0)' overwrites legacy SUMPRODUCT formulas
    // that summed TRANSACTIONS!E (which made dashboard go negative). Without
    // this, the formula update only takes effect on a fresh sheet — but most
    // admins are running on existing sheets with the broken formula.
    try { _restoreInventoryFormulas(inv); } catch(e){}

    var delLoss = _num(d.delLoss);
    if(delLoss > 0){
      var exp=ss.getSheetByName("EXPENSES");
      if(exp){
        var next=_getActualLastRow(exp,1)+1;
        _ensureRows(exp, next);
        exp.getRange(next,1,1,5).setValues([[new Date(),"Delivery Loss",d.prod+" Return",delLoss,d.nt||""]]);
      }
    }

    notifyCloudflare();
    return{ok:true,success:true};
  }catch(e){return{ok:false,msg:e.message};}
}

function generateMonthlyReport(){
  try{
    var ss=_ss();
    var tx=ss.getSheetByName("TRANSACTIONS");
    var ad=ss.getSheetByName("AD_TRACKER");
    var ord=ss.getSheetByName("ORDERS");
    var rep=ss.getSheetByName("MONTHLY_REPORT");
    if(!rep)return;
    rep.getRange(2,1,200,6).clearContent();
    var months={};
    if(tx){
      var lr=_getActualLastRow(tx,1);
      if(lr>=2){
        var rows=tx.getRange(2,1,lr-1,8).getValues();
        rows.forEach(function(r){
          if(!r[0])return;
          var d=new Date(r[0]);
          var key=d.getFullYear()+"-"+("0"+(d.getMonth()+1)).slice(-2);
          if(!months[key])months[key]={rev:0,cost:0,ad:0,ord:0};
          if(r[2]==="Sale"){months[key].rev+=_num(r[5]);months[key].cost+=_num(r[6]);}
          else if(r[2]==="Return"){months[key].rev-=_num(r[5]);months[key].cost-=_num(r[6]);}
        });
      }
    }
    if(ad){
      var alr=_getActualLastRow(ad,1);
      if(alr>=2){
        var arows=ad.getRange(2,1,alr-1,7).getValues();
        arows.forEach(function(r){
          if(!r[0])return;
          var d=new Date(r[0]);
          var key=d.getFullYear()+"-"+("0"+(d.getMonth()+1)).slice(-2);
          if(!months[key])months[key]={rev:0,cost:0,ad:0,ord:0};
          months[key].ad+=_num(r[2]);
        });
      }
    }
    if(ord){
      var olr=_getActualLastRow(ord,1);
      if(olr>=2){
        var orows=ord.getRange(2,1,olr-1,1).getValues();
        orows.forEach(function(r){
          if(!r[0])return;
          var d=new Date(r[0]);
          var key=d.getFullYear()+"-"+("0"+(d.getMonth()+1)).slice(-2);
          if(!months[key])months[key]={rev:0,cost:0,ad:0,ord:0};
          months[key].ord+=1;
        });
      }
    }
    var keys=Object.keys(months).sort();
    var out=keys.map(function(k){var m=months[k];return [k,m.rev,m.cost,m.ad,m.rev-m.cost-m.ad,m.ord];});
    if(out.length)rep.getRange(2,1,out.length,6).setValues(out);
  }catch(e){ Logger.log('[generateMonthlyReport] error: ' + e.message); }
}

// ===== GITHUB SYNC (publishes WEBSITE_SYNC + Settings as JSON) =====
// ════════════════════════════════════════════════════════
function saveGitHubSettings(body){
  try{
    // ✅ v16.4 BUGFIX: The admin panel sends SHORT keys {t,o,r,b,p}
    // (token, owner, repo, branch, path) but this function previously read
    // only the LONG keys (body.token/repo/branch/path) → every field saved
    // blank and "Sync Now" threw "GitHub Token / Repo missing". Accept BOTH
    // shapes so the settings actually persist. (Owner `o` isn't used by
    // githubSyncNow — repo already carries "owner/repo" — so it's optional.)
    var token  = body.t      != null ? body.t      : body.token;
    var repo   = body.r      != null ? body.r      : body.repo;
    var branch = (body.b     != null ? body.b      : body.branch) || "main";
    var path   = (body.p     != null ? body.p      : body.path)   || "data.json";
    var s = {
      "GitHub Token":  String(token  == null ? "" : token),
      "GitHub Repo":   String(repo   == null ? "" : repo),
      "GitHub Branch": String(branch),

function githubSyncNow(){
  var s=_getSettingsMap();
  var token=String(_safe(s["GitHub Token"]||"")).trim();
  var repo=String(_safe(s["GitHub Repo"]||"")).trim();
  var branch=String(_safe(s["GitHub Branch"]||"main")).trim();
  var path=String(_safe(s["GitHub Path"]||"data.json")).trim();
  if(!token || !repo) throw new Error("GitHub Token / Repo missing in SETTINGS");
  // Build payload
  var data=_buildPublicData();
  var content=Utilities.base64Encode(Utilities.newBlob(JSON.stringify(data,null,2),"application/json").getBytes());
  var url="https://api.github.com/repos/"+repo+"/contents/"+encodeURI(path)+"?ref="+encodeURIComponent(branch);
  var headers={"Authorization":"Bearer "+token,"Accept":"application/vnd.github+json"};
  var sha="";
  try{
    var resp=UrlFetchApp.fetch(url,{method:"get",headers:headers,muteHttpExceptions:true});
    if(resp.getResponseCode()===200){try{var j=JSON.parse(resp.getContentText());sha=j.sha||"";}catch(e){}}
  }catch(e){}
  var body={message:"YARZ sync "+new Date().toISOString(),content:content,branch:branch};
  if(sha)body.sha=sha;
  var put=UrlFetchApp.fetch("https://api.github.com/repos/"+repo+"/contents/"+encodeURI(path),{
    method:"put",headers:headers,contentType:"application/json",
    payload:JSON.stringify(body),muteHttpExceptions:true
  });
  if(put.getResponseCode()>=300) throw new Error("GitHub: "+put.getContentText());
  return true;
}

// ⛔ RATE LIMITER: 100 req/min per API key using CacheService (IP not exposed in GAS)
// Also supports per-phone rate limiting for place_order to prevent phone abuse.
function _rateLimit(key, phone){
  var cache = CacheService.getScriptCache();
  var now = Date.now();
  var minute = Math.floor(now / 60000);
  // Per-key rate limit (100/min)
  var cacheKey = "rl_" + key + "_" + minute;
  var count = parseInt(cache.get(cacheKey) || "0", 10);
  if(count >= 100) return false;
  cache.put(cacheKey, String(count + 1), 120);
  // Per-phone rate limit (5/min) — defense in depth
  if(phone){
    var phoneKey = "rl_ph_" + phone + "_" + minute;
    var phCount = parseInt(cache.get(phoneKey) || "0", 10);
    if(phCount >= 5) return false;
    cache.put(phoneKey, String(phCount + 1), 120);
  }
  return true;
}

function _placeWebsiteOrder(o){
  // ⚠️ OTP NOT IMPLEMENTED: This system has no SMS gateway integration.
  // Phone verification relies on client-side fraud detection (shield.js).
  // Server-side phone format validation is performed below as a minimum guard.
  // TODO: Integrate an SMS OTP service (e.g. Twilio, Vonage, BD SMS gateway)
  //       and add OTP verification before accepting the order.
  
  // ✅ Server-side phone format validation (minimum guard)
  var _phoneRaw = _str(o.phone || "").replace(/[^0-9]/g, "");
  if (_phoneRaw.length < 10) {
    return _webErr_("Invalid phone number — must be a valid Bangladeshi number", 400);
  }
  // Normalize to BD format for consistent storage
  if (_phoneRaw.indexOf("880") === 0) _phoneRaw = _phoneRaw.substring(3);
  if (_phoneRaw.charAt(0) === "0") _phoneRaw = _phoneRaw.substring(1);
  if (_phoneRaw.length > 10) _phoneRaw = _phoneRaw.substring(_phoneRaw.length - 10);
  if (_phoneRaw.length !== 10 || _phoneRaw.charAt(0) !== "1") {
    return _webErr_("Invalid Bangladeshi phone number", 400);
  }
  o.phone = "01" + _phoneRaw.substring(1); // store normalized form

  // ✅ v1.0 Fortress: Server-side device blocklist check
  try {
    var _devId = _str(o.deviceId || "").trim();
    if (_devId && _isDeviceBlocked(_devId)) {
      _fortressLogEvent({
        deviceId: _devId,
        phoneHash: _hashPhoneServerSafe(_str(o.phone || "")),
        ip: _str(o.ipAddress || ""),
        eventType: "shadow_ban_server",
        reason: "device_in_blocklist",
        riskScore: 100,
        actor: "auto"
      });
      return _webJson_({
        success: true, ok: true, orderId: _str(o.orderId || ("WEB-"+Date.now())),
        shadowBan: true,
        message: "Order received",
        timestamp: Date.now()
      });
    }
  } catch(_fbCheck) { Logger.log('[placeOrder] FB CAPI check error: ' + _fbCheck.message); }
  var lock = LockService.getScriptLock();
  try{
    lock.waitLock(30000);
    var ss = _ss();
    var sh = ss.getSheetByName("Website_Orders");
    if(!sh) sh=ss.insertSheet("Website_Orders");
    var inv = ss.getSheetByName("INVENTORY");
    var invLr = inv ? _getActualLastRow(inv, 1) : 0;
    var invNames = (inv && invLr >= 2) ? _flat(inv.getRange(2, 1, invLr - 1, 1).getValues()) : [];
    
    var items = o.cartItems ? o.cartItems : [o];
    // ✅ v15.45 RELIABILITY: Empty cart guard. Previously an order with no
    // items would still write a single placeholder row + decrement no
    // inventory, polluting the sheet. Reject early.

    // freeShipThreshold: 99999, freeShipSavings: 9999` to pollute the Notes
    // column and Telegram channel. Cart prices come from `items[i].price`
    // (already trusted in this function for revenue calc), so re-deriving

    if (o.freeShipApplied) {

      if (serverThreshold > 0 && serverSubtotal >= serverThreshold) {

        // ✅ v15.49 FREE-SHIP ADVANCE: When customer paid the ৳100 advance

      }
    }

        orderId,                              /* A: orderId */
        date,                                 /* B: date */

        phone,                                /* D: phone */
        _v96Addr(fullAddr, custCity),         /* E: Address */

        _str(item.product),                   /* G: Product */
        _str(item.size),                      /* H: Size */
        qty,                                  /* I: Qty */
        price,                                /* J: Price */
        dlv,                                  /* K: Delivery */
        rowTotal,                             /* L: Total */
        payment,                              /* M: Payment */
        notes,                                /* N: Notes */

        "Pending",                            /* P: Status */
        "",                                   /* Q: Courier */
        bdTime,                               /* R: Updated */
        "Order placed @ "+bdTime,             /* S: Activity */
        _str(o.deviceId || ""),               /* T: Device ID */
        _str(o.ip || ""),                     /* U: IP */
        _str(o.country || ""),                /* V: Country */
        _devInfo || "Unknown",                /* W: ASN / Device Info */
        _int(o.riskScore || 0),               /* X: Risk Score */
        _str(o.riskSignals || "")             /* Y: Risk Signals */
      ]);

      // ✅ Auto-stock & Transaction logic
      if (invNames.length) {
        var idx = invNames.indexOf(_str(item.product));
        if (idx > -1) {
          var soldCol = {S:COL.SOLD_S,M:COL.SOLD_M,L:COL.SOLD_L,XL:COL.SOLD_XL,XXL:COL.SOLD_XXL,"3XL":COL.SOLD_3XL,ONE:COL.SOLD_M}[_str(item.size).toUpperCase()];
          if (soldCol) {
            var r = idx + 2;
            var cur = _int(inv.getRange(r, soldCol).getValue());
            inv.getRange(r, soldCol).setValue(cur + qty);
            inv.getRange(r, COL.UPDATED).setValue(new Date());
            
            // Log to TRANSACTIONS so financial reports stay accurate
            var salePrice = _num(inv.getRange(r, COL.SALE).getValue());
            var costPrice = _num(inv.getRange(r, COL.COST).getValue());
            try { _logTransaction([date, _str(item.product), "Sale", _str(item.size), qty, qty * salePrice, qty * costPrice, qty * (salePrice - costPrice)]); } catch(e){}
          }
        }
      }
    }
    
    sh.getRange(next, 1, rowsToInsert.length, rowsToInsert[0].length).setValues(rowsToInsert);


    try { sh.getRange(next, 2, rowsToInsert.length, 1).setNumberFormat("yyyy-mm-dd hh:mm:ss"); } catch(e){}
    
    try { _fbCapiPurchase(o); } catch(e){}
    
    // Pass the consolidated items to telegram notification
    o.cartItems = items;
    o.total = grandTotal;
    try { _sendTelegramOrderNotification(o); } catch(e){ Logger.log('[order] Telegram notification error: ' + e.message); }
    
    try { _updateCustomerLTV(o.phone, o.customerName || '', grandTotal, new Date(), o.location || ''); } catch(e){ Logger.log('[LTV] update error: ' + e.message); }

    // ✅ v15.39 CRITICAL FIX: notifyCloudflare() is a global no-op (disabled
    // for bulk-edit storms), but a customer order is ONE discrete event that
    // absolutely needs to invalidate the storefront cache so the next visitor
    // sees real stock. Direct async purge fires a one-shot Cloudflare REST
    // call without slowing down the order response. Quota-safe — purges only
    // the products + product cache slots, nothing else.
    try { _executeCloudflarePurge(['products', 'product'], { async: true }); } catch(e){}
    return _webJson_({
      success:true, ok:true, orderId:orderId,
      timestamp: Date.now(),
      bdTime: bdTime,
      total: grandTotal, qty: items.length, 
      status: "Pending"
    });
  }catch(e){
    var isLock = e.message && e.message.toLowerCase().indexOf("lock") !== -1;
    return _webErr_(isLock ? "Heavy traffic. Please try again in 30 seconds." : e.message);
  }finally{
    if(lock) lock.releaseLock();
  }
}

// ✅ v14.0: Server-side delivery/cancellation/return CAPI events (admin-toggleable)
//   Only fires if (a) the toggle for this status is ON in admin, AND
//                (b) we haven't already fired for this orderId+status (idempotent).
//   Idempotency check: scan Activity column for "<status>_capi" marker.
function _fbCapiOrderStatusEvent(orderId, status, firstRow){
  if(!orderId || !status || !firstRow) return;
  var statusLower = String(status).toLowerCase().trim();
  // Map status to: { eventName, toggleKey, eventIdSuffix, settingKey }
  var statusMap = {
    "delivered": { event: "OrderDelivered", toggle: "pixel_evt_order_delivered", suffix: "_delivered" },
    "cancelled": { event: "OrderCancelled", toggle: "pixel_evt_order_cancelled", suffix: "_cancelled" },
    "returned":  { event: "OrderReturned",  toggle: "pixel_evt_order_returned",  suffix: "_returned"  }
  };
  var cfg = statusMap[statusLower];
  if(!cfg) return; // Unknown status — skip (we only care about terminal statuses)
  
  // Check toggle from settings sheet
  var s = _getSettingsMap();
  var toggleVal = String(s[cfg.toggle] || "").toLowerCase().trim();
  // Default OFF for status events (opt-in by admin)
  var enabled = (toggleVal === "true" || toggleVal === "1" || toggleVal === "yes" || toggleVal === "on");
  if(!enabled) return;
  
  // Idempotency: don't double-fire if order already had this status event sent
  var sh = _ss().getSheetByName("Website_Orders");
  if(!sh) return;
  var activityCell = sh.getRange(firstRow, 19);
  var activity = String(activityCell.getValue() || "");
  var firedMarker = cfg.event + "_capi_fired";
  if(activity.indexOf(firedMarker) !== -1){
    Logger.log("CAPI " + cfg.event + " already fired for " + orderId + " — skipping duplicate");
    return;
  }
  
  // Read order data from sheet — first row has all customer info + total
  var rowData = sh.getRange(firstRow, 1, 1, 19).getValues()[0];
  // Column map (1-indexed minus 1 for array): A=0, C=2 customer, D=3 phone, E=4 address,
  //   F=5 location, G=6 product, J=9 price, L=11 total, M=12 payment
  var orderIdSheet  = String(rowData[0] || orderId);
  var customerName  = String(rowData[2] || "");
  var phone         = String(rowData[3] || "");
  var addr          = String(rowData[4] || "");
  var location      = String(rowData[5] || "");
  var paymentRaw    = String(rowData[12] || "cod").toLowerCase().trim();
  
  // Aggregate total across ALL rows of this order (multi-item carts)
  var lr = _getActualLastRow(sh, 1);
  var totalValue = 0;
  if(lr >= 2){
    var allRows = sh.getRange(2, 1, lr-1, 19).getValues();
    allRows.forEach(function(r){
      if(String(r[0]) === String(orderId)){
        totalValue += (_num(r[11]) || 0); // L: Total
      }
    });
  }
  if(totalValue <= 0) totalValue = _num(rowData[11]) || 0;
  
  var customData = {
    "currency": "BDT",
    "value": totalValue,
    "order_id": orderIdSheet,
    "payment_method": paymentRaw,
    "is_prepaid": /^(bkash|nagad|rocket|bank|paid|prepaid|sslcommerz)/.test(paymentRaw) ? 1 : 0
  };
  
  var userData = {
    name: customerName,
    phone: phone,
    city: location,
    country: "BD"
  };
  
  // Send the CAPI event
  // ✅ v14.0 FIX: Only mark as fired if Facebook actually received the event.
  //   If CAPI is misconfigured or FB Graph returns 5xx, the marker is NOT
  //   written, so a re-run (e.g. after admin enters Pixel ID/Token) succeeds.
  var sent = _fbCapiSendEvent(cfg.event, orderIdSheet + cfg.suffix, customData, userData, "system_generated");
  if(!sent){
    Logger.log("CAPI " + cfg.event + " NOT sent for " + orderId + " — marker NOT written, future re-tries possible");
    return;
  }
  
  // Mark Activity column so we don't fire again
  var nowBD = Utilities.formatDate(new Date(), "Asia/Dhaka", "HH:mm:ss");
  activityCell.setValue(activity + " | " + firedMarker + " @ " + nowBD);
  Logger.log("CAPI " + cfg.event + " fired for " + orderId + " (value: " + totalValue + ")");
}

function _webDeleteWebsiteOrder(body){
  // ✅ v15.39 CRITICAL FIX: Full reversal on order delete.
  //
  // Old behavior: just deleted Website_Orders rows. INVENTORY.SOLD_<size>
  // stayed bumped from _placeWebsiteOrder, TRANSACTIONS still had the Sale
  // row → admin Dashboard kept showing revenue/profit, customer storefront
  // kept showing decremented stock. User had to manually run a Return entry
  // to undo a fake/test/cancelled order.
  //
  // New behavior mirrors saveReturnFromForm reversal pattern:
  //   1. Snapshot the order line-items BEFORE deleting (we need product/size/qty)
  //   2. For each item:
  //      a. Walk TRANSACTIONS most-recent-first → cancel matching Sale rows
  //         (full delete if qty fully covered, partial reduce otherwise)
  //      b. Decrement INVENTORY.SOLD_<size> by Math.max(0, current - qty)
  //      c. Skip if status === "Returned" (already reversed via Return form)
  //   3. _restoreInventoryFormulas (keeps AB2 = 0 source-of-truth from v15.35)
  //   4. Delete Website_Orders rows bottom-up
  //   5. Async Cloudflare purge so storefront stock restores instantly
  var lock = LockService.getScriptLock();
  try {
    lock.waitLock(20000);
    var ss = _ss();
    var sh = ss.getSheetByName("Website_Orders");
    if (!sh) return {ok:false, msg:"Website_Orders not found"};
    var lr = _getActualLastRow(sh, 1);
    if (lr < 2) return {ok:false, msg:"empty"};

    // Snapshot all rows for this order BEFORE deleting (need product/size/qty)
    var data = sh.getRange(2, 1, lr - 1, 19).getValues();
    var hits = [];
    for (var di = 0; di < data.length; di++) {
      var rr = data[di];
      if (String(rr[0]) === String(body.orderId)) {
        hits.push({
          row:     di + 2,
          product: String(rr[6] || ""),                       // G: Product
          size:    String(rr[7] || ""),                       // H: Size
          qty:     _int(rr[8]) || 1,                          // I: Qty
          status:  String(rr[15] || "").trim()                // P: Status
        });
      }
    }
    if (!hits.length) return {ok:false, msg:"not found"};

    var inv = ss.getSheetByName("INVENTORY");
    var tx  = ss.getSheetByName("TRANSACTIONS");
    // ✅ v16.1 ONE-SIZE: include the "ONE" token (sizeless products) → SOLD_M,
    // so cancelling/deleting a one-size order restores its stock just like any
    // sized order. Without this the SOLD_M restore was skipped → permanent
    // stock drift for caps/watches/etc.
    var soldColMap = {S:COL.SOLD_S, M:COL.SOLD_M, L:COL.SOLD_L, XL:COL.SOLD_XL, XXL:COL.SOLD_XXL, "3XL":COL.SOLD_3XL, ONE:COL.SOLD_M};

    // Pre-cache INVENTORY product names for fast lookup
    var iNames = [];
    var ilr = 0;
    if (inv) {
      ilr = _getActualLastRow(inv, 1);
      if (ilr >= 2) iNames = _flat(inv.getRange(2, 1, ilr - 1, 1).getValues());
    }

    for (var hi = 0; hi < hits.length; hi++) {
      var h = hits[hi];
      // Already reversed via Return form → skip to avoid double-credit
      if (h.status === "Returned" || h.status === "Cancelled") continue;

      // 1) Cancel matching Sale entries in TRANSACTIONS (mirrors saveReturnFromForm)
      if (tx && h.qty > 0) {
        var remaining = h.qty;
        var txLr = _getActualLastRow(tx, 1);
        if (txLr >= 2) {
          var txData = tx.getRange(2, 1, txLr - 1, 8).getValues();
          for (var i = txData.length - 1; i >= 0 && remaining > 0; i--) {
            var r = txData[i];
            if (String(r[1]) === h.product && String(r[2]) === "Sale" &&
                String(r[3]) === h.size && _int(r[4]) > 0) {
              var rowQty   = _int(r[4]);
              var unitRev  = _num(r[5]) / rowQty;
              var unitCost = _num(r[6]) / rowQty;
              var take     = Math.min(rowQty, remaining);
              var newQty   = rowQty - take;
              var rowIdx   = i + 2;
              if (newQty <= 0) {
                tx.deleteRow(rowIdx);
                txData = tx.getRange(2, 1, _getActualLastRow(tx,1) - 1, 8).getValues();
                i = txData.length; // becomes length-1 after i--
              } else {
                tx.getRange(rowIdx, 5).setValue(newQty);
                tx.getRange(rowIdx, 6).setValue(unitRev * newQty);
                tx.getRange(rowIdx, 7).setValue(unitCost * newQty);
                tx.getRange(rowIdx, 8).setValue((unitRev - unitCost) * newQty);
              }
              remaining -= take;
            }
          }
        }
      }

      // 2) Decrement INVENTORY.SOLD_<size> (mirrors saveReturnFromForm)
      if (inv && iNames.length) {
        var iIdx = iNames.indexOf(h.product);
        var sCol = soldColMap[_str(h.size).toUpperCase()];
        if (iIdx > -1 && sCol) {
          var ir = iIdx + 2;
          var cur = _int(inv.getRange(ir, sCol).getValue());
          inv.getRange(ir, sCol).setValue(Math.max(0, cur - h.qty));
          inv.getRange(ir, COL.UPDATED).setValue(new Date());
        }
      }
    }

    // 3) Restore inventory ARRAYFORMULA (v15.35 — keeps AB2 = 0 source-of-truth)
    try { if (inv) _restoreInventoryFormulas(inv); } catch(e){}

    // 4) Delete the Website_Orders rows bottom-up
    var rowsDesc = hits.map(function(h){ return h.row; }).sort(function(a,b){ return b - a; });
    rowsDesc.forEach(function(row){ sh.deleteRow(row); });

    // 5) Real cache purge so storefront immediately reflects restored stock
    try { _executeCloudflarePurge(['products', 'product'], { async: true }); } catch(e){}

    return {ok:true, success:true, rowsDeleted: rowsDesc.length};
  } catch(e) {
    return {ok:false, msg:e.message};
  } finally {
    if (lock) try { lock.releaseLock(); } catch(_){}
  }
}

function _webArchiveCompletedOrders(body) {
  try {
    var ss = _ss();
    var sourceSheet = ss.getSheetByName("Website_Orders");
    if (!sourceSheet) return {ok: false, msg: "Website_Orders not found"};
    
    var lr = _getActualLastRow(sourceSheet, 1);
    if (lr < 2) return {ok: true, success: true, count: 0, msg: "No orders found"};
    
    // Website_Orders has 19 columns. Status is column 16 (index 15)
    var vals = sourceSheet.getRange(1, 1, lr, 19).getValues();
    var headers = vals[0];
    
    var rowsToArchive = [];
    var rowIndicesToDelete = [];
    
    // Check from row 2 downwards
    for (var i = 1; i < vals.length; i++) {
      var row = vals[i];
      var status = String(row[15] || "").trim();
      
      // We archive Delivered, Cancelled and Returned orders
      if (status === "Delivered" || status === "Cancelled" || status === "Returned") {
        rowsToArchive.push(row);
        rowIndicesToDelete.push(i + 1); // 1-indexed sheet rows
      }
    }
    
    if (rowsToArchive.length === 0) {
      return {ok: true, success: true, count: 0, msg: "No completed orders to archive."};
    }
    
    // Ensure Archive sheet exists
    var archiveSheet = ss.getSheetByName("Website_Orders_Archive");
    if (!archiveSheet) {
      archiveSheet = ss.insertSheet("Website_Orders_Archive");
      archiveSheet.getRange(1, 1, 1, headers.length).setValues([headers]);
      archiveSheet.getRange(1, 1, 1, headers.length).setBackground("#475569").setFontColor("#FFFFFF").setFontWeight("bold");
    } else {
      _ensureColumns(archiveSheet, 19);
    }
    
    // Append rows
    var alr = _getActualLastRow(archiveSheet, 1);
    archiveSheet.getRange(alr + 1, 1, rowsToArchive.length, 19).setValues(rowsToArchive);
    
    // Delete rows bottom-up to avoid index shifting
    rowIndicesToDelete.reverse();
    for (var j = 0; j < rowIndicesToDelete.length; j++) {
      sourceSheet.deleteRow(rowIndicesToDelete[j]);

function _webUpdateSettings(body){
  try{
    var sh=_ss().getSheetByName("SETTINGS");
    if(!sh) return {ok:false,msg:"SETTINGS not found"};
    var updates=body.settings || {};
    
    // ✅ v9.7: Key alias sync — when admin saves one key name, also update the legacy alias.
    // This prevents dual-key divergence (e.g. 'Currency' vs 'Currency Symbol').
    var KEY_ALIASES = {
      "Currency":"Currency Symbol", "Currency Symbol":"Currency",
      "Contact Phone":"Store Phone", "Store Phone":"Contact Phone",
      "Contact Email":"Store Email", "Store Email":"Contact Email",
      "Business Address":"Store Address", "Store Address":"Business Address"
    };
    var aliasUpdates = {};
    Object.keys(updates).forEach(function(k){
      if(KEY_ALIASES[k] && updates[KEY_ALIASES[k]] === undefined){
        aliasUpdates[KEY_ALIASES[k]] = updates[k];
      }
    });
    Object.keys(aliasUpdates).forEach(function(k){ updates[k] = aliasUpdates[k]; });
    
    var lr=_getActualLastRow(sh,1);
    if(lr<2){
      _ensureRows(sh,2);
      sh.getRange(2,1,1,3).setValues([["Store Name","YARZ",""]]);
      lr=2;
    }
    // ✅ v9.7 PERF: Batch read → in-memory update → batch write.
    // Old approach did individual setValue() per key which was very slow (50+ API calls).
    var data=sh.getRange(2,1,lr-1,2).getValues();
    var keysToUpdate = Object.keys(updates);
    var updatedInPlace = {};
    var newRows = [];
    
    // Phase 1: Update existing rows in memory
    for(var i=0; i<data.length; i++){
      var existingKey = String(data[i][0]).trim();
      if(updates.hasOwnProperty(existingKey)){
        data[i][1] = updates[existingKey];
        updatedInPlace[existingKey] = true;
      }
    }
    // Write back all existing data in one batch
    if(data.length > 0){
      sh.getRange(2,1,data.length,2).setValues(data);
    }
    
    // Phase 2: Append new keys that didn't exist
    keysToUpdate.forEach(function(k){
      if(!updatedInPlace[k]){
        newRows.push([k, updates[k], ""]);
      }
    });
    if(newRows.length > 0) {
      var next = _getActualLastRow(sh, 1) + 1;
      _ensureRows(sh, next + newRows.length);
      sh.getRange(next, 1, newRows.length, 3).setValues(newRows);
    }
    notifyCloudflare(['store_info', 'products']);
    return {ok:true,success:true};
  }catch(e){return {ok:false,msg:e.message};}
}

// ✅ v11.7: SHA-256 helper that matches client-side normalization
function _sha256Lower(value){
  if(!value) return '';
  var v = String(value).toLowerCase().trim();
  if(!v) return '';
  var bytes = Utilities.computeDigest(Utilities.DigestAlgorithm.SHA_256, v);
  var hex = '';
  for(var i=0; i<bytes.length; i++){
    var b = (bytes[i] + 256) % 256;
    hex += ('0' + b.toString(16)).slice(-2);
  }
  return hex;
}

// ✅ v11.7: Normalize BD phone → 880XXXXXXXXX (matches client _normalizePhone)
function _normalizePhoneForCapi(p){
  if(!p) return '';
  var d = String(p).replace(/[^\d]/g, '');
  if(!d) return '';
  if(d.length === 11 && d.charAt(0) === '0') d = '88' + d;
  if(d.length === 10) d = '880' + d;
  return d;
}

// ✅ v11.7: Split full name → first + last (matches client _splitName)
function _splitNameForCapi(full){
  if(!full) return { first: '', last: '' };
  var parts = String(full).trim().split(/\s+/);
  return { first: parts[0] || '', last: parts.slice(1).join(' ') };
}

// ✅ v11.7: Generic CAPI event sender — handles ALL event types (not just Purchase)
// ✅ v14.0: Returns boolean — true if event was sent successfully, false otherwise.
//   Callers (especially _fbCapiOrderStatusEvent) use this to decide whether to
//   mark the event as "fired" — fail = no marker = future re-tries can succeed.
// ✅ v15.7 FIX: Honor admin's `pixel_net_fb_capi` master toggle. If admin
//   disables CAPI in admin panel, NO server-side CAPI events fire (Purchase,
//   ViewContent mirrors, OrderDelivered/Cancelled/Returned — all gated).
// ✅ v15.75 P0-1: Accept optional `eventSourceUrl` so each event reports the
//   ACTUAL page the user was on (Andromeda / EMQ depend on this for content
//   context). We sanitize: must be HTTPS + same hostname as our store URL,
//   else fall back to homepage. Never trust client URL blindly.
function _fbCapiSendEvent(eventName, eventId, customData, userData, actionSource, eventSourceUrl){
  try{
    var s = _getSettingsMap();
    // Master CAPI toggle — admin can disable all server-side CAPI in one switch
    var capiMasterRaw = String(s["pixel_net_fb_capi"] || "true").toLowerCase().trim();
    var capiEnabled = (capiMasterRaw !== "false" && capiMasterRaw !== "0" && capiMasterRaw !== "no" && capiMasterRaw !== "off");
    if (!capiEnabled) {
      Logger.log("CAPI " + eventName + " skipped — pix_net_fb_capi is OFF");
      return false;
    }
    var pixelId = _str(s["FB Pixel"]);
    var token = _str(s["FB CAPI Token"]);
    if(!pixelId || !token) return false; // CAPI not configured — skip silently
    // ✅ v15.44 SAFETY: Only send test_event_code if BOTH:
    //   1. Admin has set "FB CAPI Test Code" in SETTINGS, AND
    //   2. Admin has explicitly opted in via "FB CAPI Test Mode" toggle.
    //
    // Why: previously test_event_code was sent on EVERY event whenever the
    // field was non-empty. If admin entered TEST12345 to debug, then forgot
    // to clear it, ALL production events went to the Test Events tab and
    // STOPPED COUNTING for ad campaigns — silently breaking ROAS reporting
    // for days/weeks until manually noticed.
    //
    // Now: admin must flip a separate toggle on every debug session, and the
    // toggle resets to false naturally because admin clears it after testing.
    var testCode = _str(s["FB CAPI Test Code"]);
    var testModeRaw = String(s["FB CAPI Test Mode"] || "false").toLowerCase().trim();
    var testModeOn = (testModeRaw === "true" || testModeRaw === "1" || testModeRaw === "yes" || testModeRaw === "on");
    var useTestCode = testCode && testModeOn;
    if (testCode && !testModeOn) {
      Logger.log("CAPI " + eventName + ": testCode present but TEST MODE is OFF — sending as PRODUCTION (correct).");
    }

    var now = Math.floor(Date.now()/1000);

    // Build user_data with full set of matching keys
    var ud = {};
    if(userData){
      var phone = _normalizePhoneForCapi(userData.phone);
      var email = String(userData.email || '').toLowerCase().trim();
      var nm = _splitNameForCapi(userData.name || userData.customerName || '');
      var city = String(userData.city || userData.area || '').toLowerCase().replace(/\s+/g, '');
      var country = String(userData.country || 'bd').toLowerCase();
      var zp = String(userData.zip || userData.postcode || '').toLowerCase();
      var st = String(userData.state || '').toLowerCase().replace(/\s+/g, '');
      // ✅ v15.7: Add gender + dob — browser pixel sends these, CAPI was missing them.
      // Higher EMQ score = better ad attribution.
      var ge = String(userData.gender || '').toLowerCase().charAt(0); // f/m
      var db = String(userData.dob || '').replace(/[^\d]/g, ''); // YYYYMMDD

      if(email)    ud.em = [_sha256Lower(email)];
      if(phone)    ud.ph = [_sha256Lower(phone)];
      if(nm.first) ud.fn = [_sha256Lower(nm.first)];
      if(nm.last)  ud.ln = [_sha256Lower(nm.last)];
      if(city)     ud.ct = [_sha256Lower(city)];
      if(st)       ud.st = [_sha256Lower(st)];
      if(zp)       ud.zp = [_sha256Lower(zp)];
      if(country)  ud.country = [_sha256Lower(country)];
      if(ge)       ud.ge = [_sha256Lower(ge)];
      if(db)       ud.db = [_sha256Lower(db)];
      // Cookies + IP/UA from client (non-hashed)
      if(userData.fbp)  ud.fbp = String(userData.fbp);
      if(userData.fbc)  ud.fbc = String(userData.fbc);
      if(userData.externalId) ud.external_id = [_sha256Lower(String(userData.externalId))];
      if(userData.clientIp)  ud.client_ip_address = String(userData.clientIp);
      if(userData.userAgent) ud.client_user_agent = String(userData.userAgent);
    }

    var payload = {
      "data": [{
        "event_name": eventName,
        "event_time": now,
        "event_id": String(eventId),  // ✅ MUST match browser pixel for dedup
        "action_source": actionSource || "website",
        // ✅ v15.75 P0-1: Use the actual page URL passed from the browser
        //   (validated below) instead of hardcoding the homepage. Andromeda's
        //   content-context signal needs the real URL to score EMQ correctly
        //   on ViewContent / AddToCart / Purchase per-product pages.
        "event_source_url": (function(){
          var fallback = "https://" + (_str(s["Website Host"]) || "yarzclothing.xyz") + "/";
          var raw = String(eventSourceUrl || "").trim();
          if(!raw) return fallback;
          // Must be https + same hostname as our store (no open redirect surface)
          try{
            var m = raw.match(/^https:\/\/([^\/?#]+)([\/?#].*)?$/i);
            if(!m) return fallback;
            var host = m[1].toLowerCase();
            var ourHost = (fallback.match(/^https:\/\/([^\/?#]+)/i) || [])[1] || "";
            ourHost = String(ourHost).toLowerCase();
            // Allow apex + any subdomain of the store host
            if(host === ourHost || host.endsWith("." + ourHost) || ourHost.endsWith("." + host)) return raw;
          }catch(_e){}
          return fallback;
        })(),
        "user_data": ud,
        "custom_data": customData || {}
      }]
    };
    if(useTestCode) payload.test_event_code = testCode;

    var url = "https://graph.facebook.com/v22.0/" + pixelId + "/events?access_token=" + token;
    var resp = UrlFetchApp.fetch(url, {
      method: "post",
      contentType: "application/json",
      payload: JSON.stringify(payload),
      muteHttpExceptions: true
    });
    // ✅ v14.0: Inspect response code — only treat 2xx as success
    var code = resp.getResponseCode();
    if(code < 200 || code >= 300){
      Logger.log("CAPI " + eventName + " HTTP " + code + ": " + resp.getContentText());
      return false;
    }
    return true;
  }catch(e){
    Logger.log("CAPI " + eventName + " error: " + e.message);
    return false;
  }
}

// ✅ v11.7: Purchase CAPI — uses orderId (NOT orderId+'_srv') to match browser
function _fbCapiPurchase(orderData){
  try{
    if(!orderData) return;
    // ✅ v11.7: Slug helper — MUST match client _productId(p) so content_ids align.
    // Strips Bengali to stay in lockstep with storefront `slugify()` and feed `slug()`.
    // (See app.js:slugify and _fbProductFeed.slug — all three must produce the same id.)
    function _slugForCapi(name){
      return String(name||"").toLowerCase().trim()
        .replace(/[\u0980-\u09FF]+/g, '')
        .replace(/[\u0600-\u06FF]+/g, '')
        .replace(/[^a-z0-9\s-]/g, '')
        .replace(/\s+/g, '-')
        .replace(/-+/g, '-')
        .replace(/^-|-$/g, '')
        .substring(0, 80);
    }
    var customData = {
      "currency": "BDT",
      // ✅ v15.7 FIX: Use subtotal (product-only) instead of total (which includes
      // delivery charge). Reporting delivery as Purchase value inflates ROAS by
      // 5-15% and makes Facebook's value-based optimization bid against the wrong
      // number. Falls back to total if subtotal not provided (backward compat).
      "value": _num(orderData.subtotal) || (_num(orderData.total) - _num(orderData.deliveryCharge)) || _num(orderData.total) || (_num(orderData.price) * (_int(orderData.qty)||1)),
      "content_type": "product",
      "order_id": _str(orderData.orderId)
    };

    // ✅ v13.2: Bangladesh advance-payment signal (server-side)
    // Tags Purchase events with payment_method + is_prepaid so Facebook can:
    //   • Build Lookalike audiences from advance-paying customers (high LTV, no fakes)
    //   • Use value-based optimization with prepaid as quality signal
    //   • Distinguish ~30-40% COD junk from real authentic buyers
    var pmRaw = String(orderData.payment || 'cod').toLowerCase().trim();
    customData.payment_method = pmRaw;
    customData.is_prepaid = /^(bkash|nagad|rocket|bank|paid|prepaid|sslcommerz)/.test(pmRaw) ? 1 : 0;

    // Build contents + content_ids from cartItems if available, fallback to single product
    if(Array.isArray(orderData.cartItems) && orderData.cartItems.length){
      customData.contents = orderData.cartItems.map(function(it){
        return { id: _slugForCapi(it.product), quantity: _int(it.qty)||1, item_price: _num(it.price) };
      });
      customData.content_ids = orderData.cartItems.map(function(it){ return _slugForCapi(it.product); });
      customData.num_items = orderData.cartItems.length;
    } else {
      customData.contents = [{ id: _slugForCapi(orderData.product), quantity: _int(orderData.qty)||1, item_price: _num(orderData.price) }];
      customData.content_ids = [_slugForCapi(orderData.product)];
      customData.content_name = _str(orderData.product);
      customData.num_items = 1;
    }
    // ✅ v15.74: 2026 Andromeda new-customer signal — checks ORDERS sheet to see
    // if this phone (or email) has any prior order. Required for Advantage+
    // Sales campaigns to optimize for NEW customer acquisition vs existing.
    // Per Meta docs (2025-2026): customer_segmentation enum on Purchase events.
    customData.content_type = "product"; // ✅ v15.74: catalog matching
    try {
      var phoneNorm = String(orderData.phone || '').replace(/[^\d]/g, '');
      var emailNorm = String(orderData.email || '').toLowerCase().trim();
      var ordSheet = _ss().getSheetByName("ORDERS");
      var isReturning = false;
      if (ordSheet && (phoneNorm || emailNorm)) {
        var lastRow = ordSheet.getLastRow();
        if (lastRow > 1) {
          // Read just phone + email columns. Header order: orderId(1), date(2), name(3), phone(4), email(5), ...
          var data = ordSheet.getRange(2, 1, lastRow - 1, 6).getValues();
          for (var ri = 0; ri < data.length; ri++) {
            var row = data[ri];
            if (String(row[0]) === String(orderData.orderId)) continue; // skip the just-inserted order itself
            var rowPhone = String(row[3] || '').replace(/[^\d]/g, '');
            var rowEmail = String(row[4] || '').toLowerCase().trim();
            if ((phoneNorm && rowPhone && rowPhone === phoneNorm) ||
                (emailNorm && rowEmail && rowEmail === emailNorm)) {
              isReturning = true;
              break;
            }
          }
        }
      }
      customData.customer_segmentation = isReturning ? "existing_customer_to_business" : "new_customer_to_business";
      customData.order_id = String(orderData.orderId || ''); // ✅ v15.74: official Meta param
    } catch (e) {
      Logger.log("customer_segmentation lookup err: " + e.message);
      // Don't break the Purchase event if segmentation check fails
    }

    var userData = {
      name: orderData.customerName || orderData.customer || orderData.name,
      phone: orderData.phone,
      email: orderData.email,
      city: orderData.city,
      state: orderData.state,
      zip: orderData.zip || orderData.postcode,
      country: orderData.country || 'BD',
      fbp: orderData.fbp,
      fbc: orderData.fbc,
      externalId: orderData.externalId,
      clientIp: orderData.clientIp,
      userAgent: orderData.userAgent
    };

    // ✅ event_id = orderId (matches browser pixel exactly — no _srv suffix)
    _fbCapiSendEvent("Purchase", _str(orderData.orderId), customData, userData, "website");

// ===== ✅ v11.7: Facebook Product Catalog Feed (CSV) =====
// Returns CSV that Facebook Commerce Manager can ingest as a Data Source.
// Used for DABA (Dynamic Ads for Broad Audiences) — retargeting + prospecting
// with the SAME content_ids that pixel.js sends → exact catalog match.
// Spec: https://www.facebook.com/business/help/120325381656392
// ════════════════════════════════════════════════════════
function _fbProductFeed(e){
  try{
    var data = _buildPublicData();
    var s = _getSettingsMap();
    var products = data.products || [];
    var storeBase = _str(s["Store URL"]) || "https://yarzclothing.xyz";
    storeBase = storeBase.replace(/\/+$/,'');
    var currency = "BDT";

    // CSV headers per FB catalog spec (minimum required + recommended)
    var headers = [
      "id","title","description","availability","condition","price","sale_price",
      "link","image_link","brand","google_product_category","product_type",
      "additional_image_link","gender","age_group"
    ];
    var rows = [headers.join(",")];

    function csv(v){
      if(v === null || v === undefined) return "";
      var str = String(v).replace(/"/g, '""').replace(/\r?\n/g, ' ');
      return '"' + str + '"';
    }

    function slug(name){
      // ⚠️ MUST match storefront slugify() and pixel.js _slug() so feed ids align
      // with deep-link URLs and pixel content_ids. Strips Bengali to follow the
      // existing storefront router (which strips Bengali for backward compat).
      return String(name||"").toLowerCase().trim()
        .replace(/[\u0980-\u09FF]+/g, '')
        .replace(/[\u0600-\u06FF]+/g, '')
        .replace(/[^a-z0-9\s-]/g, '')
        .replace(/\s+/g, '-')
        .replace(/-+/g, '-')
        .replace(/^-|-$/g, '')
        .substring(0, 80);
    }

    function stripTags(str){
      return String(str||'').replace(/<[^>]+>/g, ' ').replace(/\s+/g, ' ').trim();
    }

    products.forEach(function(p){
      // Stock check — at least one size in stock = "in stock"
      var totalStock = (p.stockS||0)+(p.stockM||0)+(p.stockL||0)+(p.stockXL||0)+(p.stockXXL||0)+(p.stock3XL||0);
      var availability = totalStock > 0 ? "in stock" : "out of stock";

      var price = _num(p.regular) || _num(p.sale);
      var salePrice = _num(p.sale);
      var hasDiscount = salePrice > 0 && salePrice < price;
      var priceStr = (hasDiscount ? price : (salePrice || price)).toFixed(2) + " " + currency;
      var salePriceStr = hasDiscount ? (salePrice.toFixed(2) + " " + currency) : "";

      // ID = slug(name) — MUST match the content_ids sent by pixel.js (_productId)
      var id = slug(p.name);
      if(!id) return;

      // ✅ v11.7: Use hash-based product route (#product/<id>) — storefront router
      // matches `hash.indexOf('#product/')`. Query-param `?p=` would 404 → wasted ad spend.
      var link = storeBase + "/#product/" + encodeURIComponent(id);
      var img = p.image1 || p.image2 || "";
      var addlImgs = [p.image2, p.image3, p.image4, p.image5, p.image6]
        .filter(function(x){ return x && x !== img; })
        .slice(0, 5)
        .join(",");

      rows.push([
        csv(id),
        csv(p.name),
        csv(stripTags(p.description) || p.name),
        csv(availability),
        csv("new"),
        csv(priceStr),
        csv(salePriceStr),
        csv(link),
        csv(img),
        csv(_str(s["Store Name"]) || "YARZ"),
        csv("Apparel & Accessories > Clothing"),
        csv(p.category || "Apparel"),
        csv(addlImgs),
        csv("male"),
        csv("adult")
      ].join(","));
    });

    var csvText = rows.join("\n");
    return ContentService.createTextOutput(csvText)
      .setMimeType(ContentService.MimeType.CSV);
  }catch(err){
    return _webErr_("Feed error: " + err.message, 500);
  }
}

// ✅ v15.11: Self-healing webhook. Throttled to once per hour.
// Verifies the Telegram bot's webhook URL matches the current Apps Script
// deployment. If mismatched (e.g. after a redeploy with new URL), auto-fixes
// it. If not set at all, sets it. This eliminates the "I redeployed and now
// my Confirm buttons don't work" failure mode entirely — admin never has to
// manually run setupTelegramWebhook.
function _tgEnsureWebhookCurrent(token) {
  try {
    var props = PropertiesService.getScriptProperties();
    var lastCheck = parseInt(props.getProperty(TG_WEBHOOK_LAST_CHECK_KEY) || '0', 10);
    var now = Date.now();
    if (now - lastCheck < TG_WEBHOOK_CHECK_INTERVAL_MS) return; // throttled

    var deployUrl = ScriptApp.getService().getUrl();
    if (!deployUrl) return; // no deployment yet

    var infoResp = UrlFetchApp.fetch(
      "https://api.telegram.org/bot" + token + "/getWebhookInfo",
      { muteHttpExceptions: true }
    );
    var info;try{info=JSON.parse(infoResp.getContentText());}catch(e){info={};}
    if (!info || !info.ok) {
      _tgRecordDiag('error', 'getWebhookInfo failed: ' + infoResp.getContentText().slice(0, 100));
      return;
    }

    var currentWebhook = (info.result && info.result.url) || '';
    if (currentWebhook === deployUrl) {
      // Already correct — just update the throttle timestamp
      props.setProperty(TG_WEBHOOK_LAST_CHECK_KEY, String(now));
      return;
    }

    // Mismatch — auto-fix
    Logger.log('[Telegram] Webhook auto-heal: was "' + currentWebhook + '", setting to "' + deployUrl + '"');
    var setResp = UrlFetchApp.fetch(
      "https://api.telegram.org/bot" + token + "/setWebhook",
      {
        method: 'post',
        contentType: 'application/json',
        payload: JSON.stringify({ url: deployUrl }),
        muteHttpExceptions: true
      }
    );
    var setRes;try{setRes=JSON.parse(setResp.getContentText());}catch(e){setRes={};}
    if (setRes && setRes.ok) {
      _tgRecordDiag('ok', 'Webhook auto-healed to current deployment');
      props.setProperty(TG_WEBHOOK_LAST_CHECK_KEY, String(now));
    } else {
      _tgRecordDiag('error', 'Webhook auto-heal failed: ' + setResp.getContentText().slice(0, 100));
    }
  } catch (e) {
    Logger.log('[Telegram] _tgEnsureWebhookCurrent error: ' + e.message);
  }
}

function _tgApi(token, method, payload){
  try{
    var resp = UrlFetchApp.fetch("https://api.telegram.org/bot" + token + "/" + method, {
      method:"post", contentType:"application/json",
      payload:JSON.stringify(payload), muteHttpExceptions:true
    });
    var code = resp.getResponseCode();
    if (code < 200 || code >= 300) {
      var body = resp.getContentText();
      Logger.log("[Telegram] " + method + " HTTP " + code + ": " + body.slice(0, 200));
      _tgRecordDiag('error', method + ' HTTP ' + code + ' — ' + body.slice(0, 100));
      return { ok: false, code: code, body: body };
    }
    return { ok: true, code: code };
  }catch(e){
    Logger.log("[Telegram] " + method + " FETCH error: " + e.message);
    _tgRecordDiag('error', method + ' fetch error: ' + e.message);
    return { ok: false, error: e.message };
  }
}

        _tgOrderAction(botToken, chatId, msgId, cb.id, data.substring(10), "Delivered", "✅ ডেলিভারড", origText);
      } else if(data.indexOf("cancel:") === 0){
        _tgOrderAction(botToken, chatId, msgId, cb.id, data.substring(7), "Cancelled", "❌ বাতিল", origText);
      }
      return;
    }

    // --- Message Commands ---
    if(update.message){
      var msg = update.message;
      if(msg.from.id !== TG_OWNER_ID){
        _tgSend(botToken, msg.chat.id, "⛔ <b>অনুমতি নেই!</b>\nএই বটটি শুধুমাত্র YARZ Admin ব্যবহার করতে পারবেন।");
        return;
      }
      var txt = (msg.text || "").trim().toLowerCase();
      if(txt === "/start"){
        _tgSend(botToken, msg.chat.id,
          "🛒 <b>YARZ Order Bot</b>\n\n" +
          "📋 <b>কমান্ড সমূহ:</b>\n" +
          "/orders — পেন্ডিং অর্ডার\n" +
          "/confirmed — কনফার্মড অর্ডার\n" +
          "/today — আজকের সকল অর্ডার\n" +
          "/stats — আজকের সারসংক্ষেপ\n\n" +
          "✅ অর্ডার আসলে অটো নোটিফিকেশন পাবেন এবং\nবাটনে ক্লিক করে স্ট্যাটাস আপডেট করতে পারবেন।"
        );
      } else if(txt === "/orders"){
        _tgShowOrders(botToken, msg.chat.id, "Pending");
      } else if(txt === "/confirmed"){
        _tgShowOrders(botToken, msg.chat.id, "Processing");
      } else if(txt === "/today"){
        _tgShowTodayOrders(botToken, msg.chat.id);
      } else if(txt === "/stats"){
        _tgShowStats(botToken, msg.chat.id);
      }
      return;
    }
  }catch(e){
    Logger.log("TG webhook error: " + e.message);
  }
  return;
}

// ===== Order Status Update from Telegram =====
function _tgOrderAction(token, chatId, msgId, cbId, orderId, newStatus, label, origMsgText){
  try{
    var sh = _ss().getSheetByName("Website_Orders");
    if(!sh){ _tgAnswer(token, cbId, "❌ Sheet পাওয়া যায়নি"); return; }
    var lr = _getActualLastRow(sh,1);
    if(lr < 2){ _tgAnswer(token, cbId, "❌ অর্ডার পাওয়া যায়নি"); return; }
    var ids = _flat(sh.getRange(2,1,lr-1,1).getValues());
    // ✅ v14.0 FIX: Multi-item carts span N rows — update ALL of them.
    //   Previous code used ids.indexOf(orderId) which only finds the first row,
    //   leaving subsequent rows with stale status. Now matches admin panel behavior.
    var hits = [];
    ids.forEach(function(id, i){ if(String(id) === String(orderId)) hits.push(i + 2); });
    if(!hits.length){ _tgAnswer(token, cbId, "❌ অর্ডার " + orderId + " পাওয়া যায়নি"); return; }
    var row = hits[0]; // first row — used for CAPI lookup (customer fields are same on all rows)
    
    var bdNow = Utilities.formatDate(new Date(), "Asia/Dhaka", "yyyy-MM-dd HH:mm:ss");
    // Update Sheet for EVERY matching row: Status (P=16), Updated (R=18), Activity (S=19)
    hits.forEach(function(r){
      sh.getRange(r, 16).setValue(newStatus);
      sh.getRange(r, 18).setValue(bdNow);
      var oldAct = _str(sh.getRange(r, 19).getValue());
      sh.getRange(r, 19).setValue(oldAct + " | " + newStatus + " @ " + bdNow);
    });

    // ✅ v14.0: Fire Facebook CAPI status event from Telegram update path too
    try { _fbCapiOrderStatusEvent(orderId, newStatus, row); } catch(e){
      Logger.log("CAPI from Telegram path err: " + e.message);
    }

    // Show popup confirmation
    _tgAnswer(token, cbId, label + " — " + orderId);

    // Build next-step buttons
    var bdTimeNow = Utilities.formatDate(new Date(), "Asia/Dhaka", "dd MMM yyyy, hh:mm a");
    var nextBtns = [];
    if(newStatus === "Processing"){
      nextBtns = [[{text:"🚚 শিপড করুন", callback_data:"shipped:" + orderId}],
                  [{text:"✅ ডেলিভারড", callback_data:"delivered:" + orderId}]];
    } else if(newStatus === "Shipped"){
      nextBtns = [[{text:"✅ ডেলিভারড", callback_data:"delivered:" + orderId}]];
    }
    // Delivered/Cancelled = no more buttons

    // Edit original message to show updated status
    var statusUpdate = "\n\n<b>" + label + "</b> — " + bdTimeNow;
    try{
      if(origMsgText && origMsgText.length > 0){
        var safeText = _tgEscape(origMsgText);
        // Replace the "কনফার্ম করতে..." line with status update
        var updated = safeText.replace(/⏳[^\n]*/, "<b>" + label + "</b> — " + bdTimeNow)
                              .replace(/কনফার্ম করতে[^\n]*/, "<b>" + label + "</b> — " + bdTimeNow);
        if(updated === safeText) updated = safeText + statusUpdate;
        
        _tgEdit(token, chatId, msgId, updated, nextBtns.length ? {inline_keyboard:nextBtns} : {inline_keyboard:[]});
      } else {
        // Fallback — send as new message
        _tgSend(token, chatId, "<b>" + label + "</b> — <code>" + orderId + "</code> — " + bdTimeNow,
          nextBtns.length ? {inline_keyboard:nextBtns} : undefined);
      }
    }catch(e){
      // Final fallback
      _tgSend(token, chatId, label + " — <code>" + orderId + "</code> — " + bdTimeNow);
    }
  }catch(e){
    _tgAnswer(token, cbId, "❌ Error: " + e.message);
  }
}

// ===== Show Orders by Status =====
function _tgShowOrders(token, chatId, status){
  try{
    var sh = _ss().getSheetByName("Website_Orders");
    if(!sh){ _tgSend(token, chatId, "❌ Sheet পাওয়া যায়নি"); return; }
    var lr = _getActualLastRow(sh,1);
    if(lr < 2){ _tgSend(token, chatId, "📭 কোনো অর্ডার নেই"); return; }
    var data = sh.getRange(2,1,lr-1,19).getValues();
    var filtered = data.filter(function(r){ return _str(r[15]) === status; });
    if(filtered.length === 0){
      var emoji = status === "Pending" ? "📭" : "📋";
      _tgSend(token, chatId, emoji + " কোনো <b>" + status + "</b> অর্ডার নেই");
      return;
    }
    // Show latest 10
    var latest = filtered.slice(-10).reverse();
    var emoji = status === "Pending" ? "⏳" : (status === "Processing" ? "✅" : "📋");
    var msg = emoji + " <b>" + status + " অর্ডার (" + filtered.length + " টি)</b>\n\n";
    latest.forEach(function(r, i){
      var oid = _str(r[0]);
      var cust = _tgEscape(_str(r[2]));
      var phone = _str(r[3]);
      var prod = _tgEscape(_str(r[6]));
      var sz = _tgEscape(_str(r[7]));
      var total = _num(r[11]);
      var pay = _str(r[12]);
      var dt = "";
      try{ dt = Utilities.formatDate(new Date(r[1]), "Asia/Dhaka", "dd MMM, hh:mm a"); }catch(e){ dt = _str(r[1]); }
      msg += "<b>" + (i+1) + ". " + prod + "</b> (" + sz + ")\n" +
        "   📦 <code>" + oid + "</code>\n" +
        "   👤 " + cust + " | 📱 <code>" + phone + "</code>\n" +
        "   💵 ৳" + total + " | " + pay + " | " + dt + "\n\n";
    });
    if(filtered.length > 10) msg += "<i>...এবং আরও " + (filtered.length - 10) + " টি</i>";

    // Add confirm buttons for pending orders
    var btns = [];
    if(status === "Pending"){
      latest.slice(0,5).forEach(function(r){
        btns.push([{text:"✅ " + _str(r[0]).slice(-8), callback_data:"confirm:" + _str(r[0])}]);
      });
    }
    _tgSend(token, chatId, msg, btns.length ? {inline_keyboard:btns} : undefined);
  }catch(e){
    _tgSend(token, chatId, "❌ Error: " + e.message);
  }
}

function _tgShowStats(token, chatId){
  try{
    var sh = _ss().getSheetByName("Website_Orders");
    if(!sh){ _tgSend(token, chatId, "❌ Sheet পাওয়া যায়নি"); return; }

      if(st !== "Cancelled") rev += _num(r[11]);

    var msg = "📊 <b>আজকের সারসংক্ষেপ</b>\n" +

      "✅ <b>কনফার্মড:</b> " + confirmed + " টি\n" +
      "🚚 <b>শিপড:</b> " + shipped + " টি\n" +
      "📦 <b>ডেলিভারড:</b> " + delivered + " টি\n";
    if(cancelled > 0) msg += "❌ <b>বাতিল:</b> " + cancelled + " টি\n";

      "💰 <b>আজকের রেভিনিউ:</b> ৳" + rev + "\n\n" +
      "🕐 " + Utilities.formatDate(new Date(), "Asia/Dhaka", "dd MMM yyyy, hh:mm a");
    _tgSend(token, chatId, msg);
  }catch(e){
    _tgSend(token, chatId, "❌ Error: " + e.message);
  }

/**
 * ✅ Run this function ONCE from Apps Script Editor to connect your Telegram Bot.
 * Steps: 
 *   1. Open Apps Script Editor
 *   2. Select this function "setupTelegramWebhook" from the dropdown
 *   3. Click ▶ Run
 *   4. Check Execution Log — it should say "Webhook set successfully!"
 */
function setupTelegramWebhook() {
  var botToken = TG_BOT_TOKEN;
  if (!botToken) {
    Logger.log("❌ ERROR: 'TG Bot Token' is empty in your Settings sheet!");
    Logger.log("➡️ Go to your Google Sheet → Settings tab → Add your bot token under 'TG Bot Token'");
    return;
  }
  
  // Use the current deployment URL as the webhook
  var deployUrl = ScriptApp.getService().getUrl();
  Logger.log("📡 Setting webhook to: " + deployUrl);
  
  var response = UrlFetchApp.fetch(
    "https://api.telegram.org/bot" + botToken + "/setWebhook", {
      method: "post",
      contentType: "application/json",
      payload: JSON.stringify({ url: deployUrl }),
      muteHttpExceptions: true
    }
  );
  
  var result;try{result=JSON.parse(response.getContentText());}catch(e){result={};}
  if (result.ok) {
    Logger.log("✅ Webhook set successfully! Your bot is now connected.");
    Logger.log("📱 Open Telegram → Send /start to your bot → You should get a welcome message.");
  } else {
    Logger.log("❌ Webhook setup failed: " + JSON.stringify(result));
    Logger.log("💡 Make sure your bot token is correct. Get it from @BotFather on Telegram.");
  }
}

/**
 * Order cleanup — runs nightly via the time-based trigger (setupDailyTrigger).
 *
 *   • Website_Orders → deleted after 90 DAYS (~3 months). High volume,
 *     customer-facing; also expires from the customer's browser cache on
 *     the same window (yarz_my_orders PII TTL = 90 days).
 *   • ORDERS (manual)→ deleted after 6 MONTHS (MANUAL_RETENTION_DAYS = 180).
 *     These are the orders YOU enter by hand from Messenger / Facebook, so
 *     they're kept far longer for reference. To keep them ONE YEAR instead,
 *     change MANUAL_RETENTION_DAYS below from 180 to 365.
 *
 * IMPORTANT: This ONLY removes order ROWS so the lines free up. The money /
 * accounting lives in the TRANSACTIONS sheet + MONTHLY_REPORT + YEARLY_REPORT,
 * which are NEVER touched here — so revenue/profit history stays permanent
 * even after an order row is deleted.
 */
function cleanupOldOrders() {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();

    // ── 1. Website orders: 90 days ── (Date = Col B / index 2)
    //     ✅ v17.15: bumped 30→90 days to match yarz_my_orders PII TTL.
    var webCutoff = new Date();
    webCutoff.setDate(webCutoff.getDate() - 90);
    _pruneOrdersByDate(ss.getSheetByName("Website_Orders"), 2, webCutoff);

    // ── 2. Manual ORDERS: 6 months ── (Date = Col A / index 1)
    //    Change 180 → 365 to keep manual orders for a full year instead.
    var MANUAL_RETENTION_DAYS = 180;
    var manualCutoff = new Date();
    manualCutoff.setDate(manualCutoff.getDate() - MANUAL_RETENTION_DAYS);
    _pruneOrdersByDate(ss.getSheetByName("ORDERS"), 1, manualCutoff);

    return "Cleanup Successful";
  } catch(e) {
    return "Cleanup Failed: " + e.message;
  }
}

/**
 * Deletes every row whose date (column `dateCol`, 1-based) is older than
 * `cutOffDate`. Rows are removed bottom-up in CONTIGUOUS BATCHES (single
 * deleteRows() per block) so the job stays fast and well under the 6-minute
 * Apps Script limit even if the trigger missed several days and tens of
 * thousands of old rows must be cleared in one run. Returns rows deleted.
 */
function _pruneOrdersByDate(sh, dateCol, cutOffDate) {
  if (!sh) return 0;
  var lr = sh.getLastRow();
  if (lr < 2) return 0;
  var dates = sh.getRange(2, dateCol, lr - 1, 1).getValues();
  var toDelete = [];
  for (var i = 0; i < dates.length; i++) {
    var d = new Date(dates[i][0]);
    if (!isNaN(d.getTime()) && d < cutOffDate) toDelete.push(i + 2); // +2: data starts at row 2
  }
  if (!toDelete.length) return 0;
  // toDelete is ascending. Walk from the bottom, grouping consecutive row
  // numbers into one deleteRows() call. Deleting the higher block first keeps
  // every lower row number still valid.
  var deleted = 0;
  var end = toDelete.length - 1;
  while (end >= 0) {
    var start = end;
    while (start > 0 && toDelete[start - 1] === toDelete[start] - 1) start--;
    var runStart = toDelete[start];
    var count = toDelete[end] - runStart + 1;
    sh.deleteRows(runStart, count);
    deleted += count;
    end = start - 1;
  }
  return deleted;
}

/**
 * Creates a Time-Driven Trigger to run cleanupOldOrders automatically every night.
 * Run this function ONCE from the Apps Script editor.
 */
function _setupAutoCleanup() {
  var triggers = ScriptApp.getProjectTriggers();
  for (var i = 0; i < triggers.length; i++) {
    if (triggers[i].getHandlerFunction() === "cleanupOldOrders") {
      ScriptApp.deleteTrigger(triggers[i]);
    }
  }
  ScriptApp.newTrigger("cleanupOldOrders")
    .timeBased()
    .everyDays(1)
    .atHour(1) // Runs at 1 AM
    .create();
}

// KEEP-ALIVE PING — eliminates GAS cold-start latency for customers
// ────────────────────────────────────────────────────────────────────
// Without this, every 6-15 min of idle time causes the next customer
// hit to wait 3-15 seconds while the GAS isolate boots. We hit our
// own Worker every 5 min, which:
//   1. Forwards a `health` action to GAS  → keeps GAS warm
//   2. Refreshes Worker isolate            → keeps Worker warm
//   3. Ensures all PREWARM_ACTIONS edge cache slots stay populated
// Run setupKeepAlivePing() ONCE from the Apps Script editor.
// ════════════════════════════════════════════════════════════════════
function healthPing() {
  try {
    var workerUrl = (typeof CLOUDFLARE_WORKER_URL === 'string' ? CLOUDFLARE_WORKER_URL : '').replace('/__purge', '');
    if (!workerUrl) return;
    UrlFetchApp.fetch(workerUrl + '/__health', {
      method: 'get',
      muteHttpExceptions: true,
      followRedirects: true,
      headers: { 'User-Agent': 'YARZ-GAS-KeepAlive/1.0' }
    });
  } catch (e) {
    Logger.log('[healthPing] ' + e.message);
  }
}

function setupKeepAlivePing() {
  var triggers = ScriptApp.getProjectTriggers();
  for (var i = 0; i < triggers.length; i++) {
    if (triggers[i].getHandlerFunction() === 'healthPing') {
      ScriptApp.deleteTrigger(triggers[i]);
    }
  }
  ScriptApp.newTrigger('healthPing')
    .timeBased()
    .everyMinutes(5)
    .create();
  Logger.log('[setupKeepAlivePing] Trigger created — runs every 5 min.');
}

function publishToCloudflareAction() {
  var ui = SpreadsheetApp.getUi();
  var res = _executeCloudflarePurge(null, { async: false });
  if (res && res.ok) {
    ui.alert("✅ Published Successfully!", "The website has been updated with your latest changes.", ui.ButtonSet.OK);
  } else {
    ui.alert("❌ Publish Failed", "Could not publish updates to Cloudflare.\n\nError: " + (res.error || res.body || "Unknown"), ui.ButtonSet.OK);
  }
}

function _executeCloudflarePurge(actions, opts) {
  opts = opts || {};
  var isAsync = opts.async !== false;
  var list = (actions && actions.length) ? actions : DEFAULT_PURGE_ACTIONS;

  var payload = JSON.stringify({ actions: list });

  var options = {
    method: 'post',
    contentType: 'application/json',
    headers: { 'X-Purge-Key': _getPurgeSecret() },
    payload: payload,
    muteHttpExceptions: true,
    followRedirects: true
  };

  if (isAsync) {
    try {
      _scheduleAsyncPurge(payload);
      return { ok: true, mode: 'async-scheduled' };
    } catch (e) { Logger.log('[purge] async schedule error: ' + e.message); }
  }

  try {
    var res = UrlFetchApp.fetch(_getCloudflareWorkerUrl(), options);
    var code = res.getResponseCode();
    var body = res.getContentText();
    Logger.log('[notifyCloudflare] ' + code + ' → ' + body);
    return { ok: code >= 200 && code < 300, code: code, body: body };
  } catch (err) {
    Logger.log('[notifyCloudflare] ERROR: ' + err.message);
    return { ok: false, error: err.message };
  }
}

function _scheduleAsyncPurge(payload) {
  var props = PropertiesService.getScriptProperties();
  props.setProperty('PENDING_PURGE_PAYLOAD', payload);
  props.setProperty('PENDING_PURGE_TS', String(Date.now()));

  var triggers = ScriptApp.getProjectTriggers();
  for (var i = 0; i < triggers.length; i++) {
    if (triggers[i].getHandlerFunction() === '_runPendingPurge') return;
  }
  ScriptApp.newTrigger('_runPendingPurge')
    .timeBased()
    .after(1)
    .create();
}

function _runPendingPurge() {
  var triggers = ScriptApp.getProjectTriggers();
  for (var i = 0; i < triggers.length; i++) {
    if (triggers[i].getHandlerFunction() === '_runPendingPurge') {
      ScriptApp.deleteTrigger(triggers[i]);
    }
  }

  var props = PropertiesService.getScriptProperties();
  var payload = props.getProperty('PENDING_PURGE_PAYLOAD');
  if (!payload) return;
  props.deleteProperty('PENDING_PURGE_PAYLOAD');
  props.deleteProperty('PENDING_PURGE_TS');

  try {
    UrlFetchApp.fetch(_getCloudflareWorkerUrl(), {
      method: 'post',
      contentType: 'application/json',
      headers: { 'X-Purge-Key': _getPurgeSecret() },
      payload: payload,
      muteHttpExceptions: true,
      followRedirects: true
    });
  } catch (e) {
    Logger.log('[_runPendingPurge] ERROR: ' + e.message);
  }
}

function _hashPhoneServerSafe(phone){
  if(!phone) return "";
  var norm = _str(phone).replace(/\D/g, "");
  // Salt is the same one used by app.js (yarz_fortress_salt).
  // We re-derive it from FORT_SALT_SCRIPT_PROP if set, else from deploy marker.
  var salt = "";
  try { salt = PropertiesService.getScriptProperties().getProperty("FORTRESS_SALT") || ""; } catch(e){}
  if (!salt) {
    // Fallback: use ADMIN_SECRET-derived salt so it stays stable per-deploy
    salt = _str(_getAdminSecret() || "").slice(0, 16);
  }
  // FNV-1a 32-bit (must match client-side _fnv1a in fortress.js)
  var h = 0x811c9dc5;
  var combined = norm + salt;
  for (var i = 0; i < combined.length; i++) {
    h ^= combined.charCodeAt(i);
    h = (h + ((h<<1) + (h<<4) + (h<<7) + (h<<8) + (h<<24))) >>> 0;
  }
  return "ph_" + ("0000000" + h.toString(16)).slice(-8);
}

function _fortressBlock(body){
  var lock = LockService.getScriptLock();
  try {
    lock.waitLock(30000);
    var deviceId = _str(body.deviceId || "").trim();
    if (!deviceId) return { ok: false, msg: "deviceId required" };
    var blockSh = _ss().getSheetByName("Blocked_Devices");
    if (!blockSh) _setupBlockedDevices(), blockSh = _ss().getSheetByName("Blocked_Devices");
    var lr = _getActualLastRow(blockSh, 1);
    // Check if already exists
    if (lr >= 2) {
      var ids = blockSh.getRange(2, 1, lr-1, 1).getValues();
      for (var i = 0; i < ids.length; i++) {
        if (_str(ids[i][0]).trim() === deviceId) {
          // Update existing row
          blockSh.getRange(i+2, 2, 1, 6).setValues([[
            new Date().toISOString(),
            _str(body.blockedBy || "admin"),
            _str(body.reason || "manual"),
            _str(body.blockType || "hard"),
            _str(body.expiresAt || ""),
            _str(body.adminNotes || "")
          ]]);
          blockSh.getRange(i+2, 8).setValue("active");
          return { ok: true, success: true, deviceId: deviceId, updated: true };
        }
      }
    }
    // Insert new row
    var next = _getActualLastRow(blockSh, 1) + 1;
    _ensureRows(blockSh, next + 1);
    blockSh.getRange(next, 1, 1, 13).setValues([[
      deviceId,
      new Date().toISOString(),
      _str(body.blockedBy || "admin"),
      _str(body.reason || "manual"),
      _str(body.blockType || "hard"),
      _str(body.expiresAt || ""),
      _str(body.adminNotes || ""),
      "active",
      new Date().toISOString(),
      _int(body.orderAttempts || 0),
      _int(body.riskScore || 0),
      _str(body.phonesSeen || ""),
      _str(body.ipsSeen || "")
    ]]);
    _fortressLogEvent({
      deviceId: deviceId,
      eventType: "block_added",
      reason: _str(body.reason || "manual"),
      actor: "admin"
    });
    return { ok: true, success: true, deviceId: deviceId, created: true };
  } catch(e){
    var isLock = e.message && e.message.toLowerCase().indexOf("lock") !== -1;
    return { ok: false, msg: isLock ? "Heavy traffic. Try again in 30s." : e.message };
  } finally {
    if(lock) lock.releaseLock();
  }
}

function snapshotMonth() {
  var d = new Date();
  d.setMonth(d.getMonth() - 1);
  var m = d.getMonth() + 1;
  var y = d.getFullYear();
  var monthId = y + '-' + (m < 10 ? '0'+m : m);
  var sh = _ensureMonthlySnapshotsTab();
  var all = sh.getDataRange().getValues();
  for(var i=1; i<all.length; i++) {
    if(all[i][0] === monthId) return "Already Snapshotted";
  }
  var prov = _computeProvisionalSnapshot(y, m);
  sh.appendRow([monthId, prov.revenue, prov.cost, prov.adSpend, prov.netProfit]);
  return "Snapshotted " + monthId;
}

function _readCurrentMonthSnapshot() {
  var d = new Date();
  return _computeProvisionalSnapshot(d.getFullYear(), d.getMonth() + 1);
}

function _computeProvisionalSnapshot(y, m) {
  var ord = _readSheet('ORDERS');
  var web = _readSheet('Website_Orders');
  var allOrd = ord.concat(web);
  var rev = 0;
  allOrd.forEach(function(o) {
    var dd = new Date(o[1]);
    if(dd.getFullYear() === y && (dd.getMonth()+1) === m && o[15] !== 'Cancelled' && o[15] !== 'Returned') {
      rev += Number(o[11]) || 0;
    }
  });
  var tx = _readSheet('TRANSACTIONS');
  var cost = 0;
  tx.forEach(function(t) {
    var dd = new Date(t[1]);
    if(dd.getFullYear() === y && (dd.getMonth()+1) === m && t[2] === 'Expense') {
      cost += Number(t[3]) || 0;
    }
  });
  var ads = _readSheet('AD_TRACKER');
  var adSpend = 0;
  ads.forEach(function(a) {
    var dd = new Date(a[0]);
    if(dd.getFullYear() === y && (dd.getMonth()+1) === m) {
      adSpend += Number(a[2]) || 0;
    }
  });
  return { revenue: rev, cost: cost, adSpend: adSpend, netProfit: rev - cost - adSpend };
}

function _newToken_() {
  var uuid = Utilities.getUuid() + Utilities.getUuid();
  return uuid.replace(/-/g, "").toLowerCase().slice(0, 64);
}

function _loginRateCheck_(identifier, ip) {
  var sh = _attemptsSheet_();
  var lr = _getActualLastRow(sh, 1);
  if (lr < 2) return true;
  var data = sh.getRange(2, 1, lr - 1, 4).getValues();
  var now = Date.now();
  var fails = 0;
  var targetTime = now - 15 * 60 * 1000; // 15 mins ago
  
  for (var i = data.length - 1; i >= 0; i--) {
    var row = data[i];
    var ts = new Date(row[1]).getTime();
    if (ts < targetTime) break;
    
    var rowId = String(row[0]);
    var rowIp = String(row[3]);
    var success = row[2];
    
    if (success === false || String(success).toLowerCase() === "false") {
      if ((ip && rowIp === ip) || (identifier && rowId === identifier)) {
        fails++;
      }
    }
  }
  return fails < 5;
}

  }
}

function _secureCompare(a, b) {
  if (typeof a !== 'string' || typeof b !== 'string') return false;
  var aLen = a.length;
  var bLen = b.length;
  if (aLen !== bLen) {
    var temp = a;
    a = b;
    b = temp;
  }
  var result = 0;
  var limit = Math.max(aLen, bLen);
  for (var i = 0; i < limit; i++) {
    var charA = a.charCodeAt(i % aLen) || 0;
    var charB = b.charCodeAt(i % bLen) || 0;
    result |= charA ^ charB;
  }
  return result === 0 && aLen === bLen;
}

function _webAdminLogin_(body) {
  try {
    var user = String(body.adminUser || "").trim();
    var pass = String(body.adminPass || "").trim();
    var ip = String(body.clientIp || "").trim();
    
    if (!_loginRateCheck_(user, ip)) {
      var lockTime = 15 * 60; // 15 mins in seconds
      return _webJson_({ success: false, locked: true, retryAfter: lockTime, error: "Too many failed attempts. Locked for 15 minutes." });
    }
    
    var expectedUser = _getAdminUsername();
    var expectedPass = _getAdminSecret();
    
    var userOk = _secureCompare(user, expectedUser);
    var passOk = _secureCompare(pass, expectedPass);
    
    if (userOk && passOk) {
      var token = _newToken_();
      var expiresAt = Date.now() + 30 * 60 * 1000; // 30 min
      
      // Save to CacheService
      var sessData = { user: user, expires: expiresAt };
      CacheService.getScriptCache().put("sess_" + token, JSON.stringify(sessData), 1800);
      
      // Save to Sheet
      var sh = _sessionsSheet_();
      sh.appendRow([token, user, new Date(), new Date(), new Date(expiresAt), false, body.userAgent || ""]);
      
      _logLoginAttempt_(user, true, ip);
      
      return _webJson_({ success: true, token: token, expiresAt: expiresAt, ttlMs: 30 * 60 * 1000 });
    } else {
      _logLoginAttempt_(user, false, ip);
      return _webJson_({ success: false, error: "ভুল ইউজারনেম অথবা পাসওয়ার্ড!" });
    }
  } catch(e) {
    return _webJson_({ success: false, error: e.message });
  }
}

function _webAdminLogout_(body) {
  try {
    var token = String(body.sessionToken || "").trim();
    if (token) {
      CacheService.getScriptCache().remove("sess_" + token);
      
      var sh = _sessionsSheet_();
      var lr = _getActualLastRow(sh, 1);
      if (lr >= 2) {
        var data = sh.getRange(2, 1, lr - 1, 1).getValues();
        for (var i = 0; i < data.length; i++) {
          if (String(data[i][0]) === token) {
            sh.getRange(i + 2, 6).setValue(true); // F column (Revoked)
            break;
          }
        }
      }
    }
    return _webJson_({ success: true });
  } catch(e) {
    return _webJson_({ success: false, error: e.message });
  }
}

function _requireAdmin_(body) {
  var token = String(body.sessionToken || "").trim();
  if (!token) return "⛔ Session token required.";
  
  var cache = CacheService.getScriptCache();
  var cached = cache.get("sess_" + token);
  if (cached) {
    try {
      var sess = JSON.parse(cached);
      if (Date.now() < sess.expires) {
        var newExpires = Date.now() + 30 * 60 * 1000;
        sess.expires = newExpires;
        cache.put("sess_" + token, JSON.stringify(sess), 1800);
        return true;
      }
    } catch(e) {}
  }
  
  var sh = _sessionsSheet_();
  var lr = _getActualLastRow(sh, 1);
  if (lr < 2) return "⛔ No active sessions found.";
  
  var data = sh.getRange(2, 1, lr - 1, 6).getValues();
  for (var i = 0; i < data.length; i++) {
    var row = data[i];
    if (String(row[0]) === token) {
      var expires = new Date(row[4]).getTime();
      var revoked = row[5];
      
      if (revoked === true || String(revoked).toLowerCase() === "true") {
        return "⛔ Session has been revoked.";
      }
      
      if (expires < Date.now()) {
        return "⛔ Session has expired.";
      }
      
      var newExpires = Date.now() + 30 * 60 * 1000;
      var sessData = { user: String(row[1]), expires: newExpires };
      cache.put("sess_" + token, JSON.stringify(sessData), 1800);
      
      sh.getRange(i + 2, 4).setValue(new Date()); // LastSeenAt
      sh.getRange(i + 2, 5).setValue(new Date(newExpires)); // ExpiresAt
      
      return true;
    }
  }
  return "⛔ Invalid session token.";
}
